bool can_attack(const queen& white, const queen& black)
{
    // Same row
    if (white.row() == black.row()) {
        return true;
    }
    // Same column
    if (white.column() == black.column()) {
        return true;
    }
    // Same diagonal
    if ((static_cast<board_index_t>((white.row() > black.row()) ? (white.row() - black.row()) : (black.row() - white.row())))
        == static_cast<board_index_t>((white.column() > black.column()) ? (white.column() - black.column()) : (black.column() - white.column())))) {
        return true;
    }
    return false;
}

        return true;
    }
    // Same column
    if (white.column() == black.column()) {
        return true;
