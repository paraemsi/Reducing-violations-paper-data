[[nodiscard]] std::chrono::day
find_day_in_range(const std::chrono::year            & year,
                  const std::chrono::month           & month,
                  const std::chrono::weekday         & target_weekday,
                  std::uint32_t                        first_day,
                  std::uint32_t                        last_day)
{
    for (std::uint32_t d { first_day }; d <= last_day; ++d) {
        const std::chrono::year_month_day current { year, month,
                                                    std::chrono::day { d } };
        if (std::chrono::weekday { current } == target_weekday) {
            return std::chrono::day { d };
        }
    }

       provided range, so reaching this line would indicate a logic error. */
    return std::chrono::day { first_day };

    : m_year { std::chrono::year { year } },

scheduler::get_day(const std::chrono::weekday & week_day,
                   Schedule                     schedule) const
{
    std::chrono::day result_day { 1U };

    switch (schedule) {
    case Schedule::first: {
        result_day = find_day_in_range(m_year, m_month, week_day, 1U, 7U);
        break;

        result_day = find_day_in_range(m_year, m_month, week_day, 8U, 14U);
