std::unordered_set<std::string> Robot::used_names_;

std::mt19937& Robot::rng() {
    static std::mt19937 gen(std::random_device{}());
    return gen;
}

std::string Robot::generate_unique_name() {
    static std::uniform_int_distribution<int> letter_dist('A', 'Z');
    static std::uniform_int_distribution<int> digit_dist(0, 9);

    std::string candidate;
    do {
        candidate.clear();
        candidate += static_cast<char>(letter_dist(rng()));
        candidate += static_cast<char>(letter_dist(rng()));
        for (int i = 0; i < 3; ++i) {
            candidate += static_cast<char>('0' + digit_dist(rng()));
        }
    } while (used_names_.find(candidate) != used_names_.end());

    return candidate;
}

Robot::Robot() = default;

const std::string& Robot::name() {
    if (name_.empty()) {
        name_ = generate_unique_name();
        used_names_.insert(name_);
    }
    return name_;
}

void Robot::reset() {
    if (!name_.empty()) {
        used_names_.erase(name_);
        name_.clear();
    }
}

}  // namespace robot_name
