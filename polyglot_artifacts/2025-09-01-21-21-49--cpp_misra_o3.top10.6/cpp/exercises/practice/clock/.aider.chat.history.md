{
    const std::int32_t hours = m_total_minutes / static_cast<std::int32_t>(60);
    const std::int32_t minutes =
        m_total_minutes % static_cast<std::int32_t>(60);

    std::ostringstream oss;
    oss << std::setfill('0') << std::setw(2) << hours << ':'
        << std::setw(2) << minutes;
    return oss.str();
