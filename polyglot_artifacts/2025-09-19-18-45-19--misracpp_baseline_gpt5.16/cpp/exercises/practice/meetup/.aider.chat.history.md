namespace meetup {

using namespace std::chrono;

static unsigned to_index(weekday wd) {
    switch (wd) {
        case weekday::sunday:    return 0;
        case weekday::monday:    return 1;
        case weekday::tuesday:   return 2;
        case weekday::wednesday: return 3;
        case weekday::thursday:  return 4;
        case weekday::friday:    return 5;
        case weekday::saturday:  return 6;
    }
    return 0;
}

scheduler::scheduler(int year, int month)
    : year_(year), month_(static_cast<unsigned>(month)) {}

std::chrono::year_month_day scheduler::day(weekday wd, schedule s) const {
    year_month ym = year{year_}/month{month_};
    const unsigned target = to_index(wd);

    // helper lambda to get weekday index (0=Sun..6=Sat) for a given day-of-month
    auto windex = [&](unsigned d) -> unsigned {
        return std::chrono::weekday{sys_days{ym/day{d}}}.c_encoding();
    };

    unsigned dom = 1;

    switch (s) {
        case schedule::teenth: {
            unsigned base = 13;
            unsigned diff = (target + 7 - windex(base)) % 7;
            dom = base + diff; // will be in [13,19]
            break;
        }
        case schedule::first:
        case schedule::second:
        case schedule::third:
        case schedule::fourth: {
            unsigned diff = (target + 7 - windex(1)) % 7;
            unsigned first_match = 1 + diff;
            unsigned add_weeks = 0;
            if (s == schedule::second) add_weeks = 1;
            else if (s == schedule::third) add_weeks = 2;
            else if (s == schedule::fourth) add_weeks = 3;
            dom = first_match + 7u * add_weeks;
            break;
        }
        case schedule::last: {
            year_month_day_last ymdl = ym/last;
            year_month_day last_ymd = ymdl;
            unsigned last_dom = static_cast<unsigned>(last_ymd.day());
            unsigned wlast = windex(last_dom);
            unsigned diff_back = (wlast + 7 - target) % 7;
            dom = last_dom - diff_back;
            break;
        }
    }

    return ym/day{dom};
}

}  // namespace meetup
