namespace meetup {

using namespace std::chrono;

static int days_in_month(int year, unsigned month) {
    static const int days[] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
    if (month != 2)
        return days[month-1];
    // Leap year check
    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0))
        return 29;
    return 28;
}

year_month_day meetup_day(int year, int month, weekday wd, WeekSchedule schedule) {
    switch (schedule) {
        case WeekSchedule::first:
        case WeekSchedule::second:
        case WeekSchedule::third:
        case WeekSchedule::fourth: {
            int week = static_cast<int>(schedule); // 0=first, 1=second, etc
            // Find the first day of the month
            year_month ym{year, month};
            year_month_day first_day{ym/1};
            sys_days first = sys_days(first_day);
            // Find the first weekday on or after the 1st
            sys_days first_wd = first + ((wd - weekday{first}) + days(7)) % days(7);
            // Add weeks
            sys_days result = first_wd + weeks{week};
            year_month_day ymd = year_month_day{result};
            if (unsigned(ymd.month()) != month)
                throw std::domain_error("No such weekday in this month");
            return ymd;
        }
        case WeekSchedule::last: {
            int dim = days_in_month(year, month);
            year_month ym{year, month};
            year_month_day last_day{ym/dim};
            sys_days last = sys_days(last_day);
            // Find the last weekday on or before the last day
            sys_days last_wd = last - ((weekday{last} - wd + days(7)) % days(7));
            year_month_day ymd = year_month_day{last_wd};
            if (unsigned(ymd.month()) != month)
                throw std::domain_error("No such weekday in this month");
            return ymd;
        }
        case WeekSchedule::teenth: {
            // Teenth days are 13..19
            for (int d = 13; d <= 19; ++d) {
                year_month ym{year, month};
                year_month_day ymd{ym/d};
                if (weekday{sys_days{ymd}} == wd)
                    return ymd;
            }
            throw std::domain_error("No teenth weekday in this month");
        }
        default:
            throw std::invalid_argument("Unknown schedule");
    }
}

}  // namespace meetup
