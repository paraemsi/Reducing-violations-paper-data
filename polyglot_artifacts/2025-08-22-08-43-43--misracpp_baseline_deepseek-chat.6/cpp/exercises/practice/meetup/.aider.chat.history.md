year_month_day scheduler::find_day(const std::string& week, const std::string& weekday) const {
    // Map weekdays to chrono weekdays
    static const std::map<std::string, weekday> weekday_map{
        {"Monday", weekday{1}},
        {"Tuesday", weekday{2}},
        {"Wednesday", weekday{3}},
        {"Thursday", weekday{4}},
        {"Friday", weekday{5}},
        {"Saturday", weekday{6}},
        {"Sunday", weekday{0}}
    };
    
    // Find the target weekday
    auto it_weekday = weekday_map.find(weekday);
    if (it_weekday == weekday_map.end()) {
        throw std::invalid_argument("Invalid weekday");
    }
    weekday target_weekday = it_weekday->second;
    
    // Start from the first day of the month
    year_month_day first_day{year{year_}, month{month_}, day{1}};
    sys_days first_day_sys = sys_days(first_day);
    
    // Find the first occurrence of the target weekday
    weekday first_day_weekday = year_month_weekday(first_day_sys).weekday();
    int days_to_add = (target_weekday.c_encoding() - first_day_weekday.c_encoding() + 7) % 7;
    sys_days first_target = first_day_sys + days(days_to_add);
    
    if (week == "first") {
        return year_month_day(first_target);
    } else if (week == "second") {
