meetup::date scheduler::day(weekday wd, schedule s) const {
    using namespace std::chrono;

    const auto desired = to_chrono_weekday(wd);
    const year y{year_};
    const month m{static_cast<unsigned>(month_)};

    if (s == schedule::teenth) {
        for (unsigned d = 13; d <= 19; ++d) {
            year_month_day ymd{y/m/day{d}};
            weekday w{sys_days{ymd}};
            if (w == desired) {
                return ymd;
            }
        }
        throw std::logic_error("No teenth weekday found (should be unreachable)");
    }

    if (s == schedule::last) {
        sys_days last_sd{y/m/last};
        weekday last_wd{last_sd};
        unsigned diff = (last_wd.c_encoding() + 7u - desired.c_encoding()) % 7u;
        return year_month_day{last_sd - days{diff}};
    }

    const unsigned n = nth_from_schedule(s);
    if (n < 1 || n > 4) {
        throw std::invalid_argument("Invalid schedule for nth occurrence");
