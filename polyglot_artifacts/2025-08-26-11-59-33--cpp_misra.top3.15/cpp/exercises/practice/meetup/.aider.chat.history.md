std::uint8_t days_in_month(meetup::year_t year, meetup::month_t month)
{
    // February leap year check
    if(month == 2U)
    {
        if(((year % 4) == 0) && (((year % 100) != 0) || ((year % 400) == 0)))
        {
            return 29U;
        }

    {
        return 30U;
    }
    // All others
    return 31U;
}

std::uint8_t calc_weekday(meetup::year_t year, meetup::month_t month, meetup::day_t day)
{
    // Zeller's congruence, 0=Saturday, 1=Sunday, ..., 6=Friday
    meetup::year_t y = year;
    meetup::month_t m = month;
    if(m < 3U)
    {
        m = static_cast<meetup::month_t>(m + 12U);
        y = static_cast<meetup::year_t>(y - 1);
    }
    meetup::day_t q = day;
    meetup::year_t K = static_cast<meetup::year_t>(y % 100);
    meetup::year_t J = static_cast<meetup::year_t>(y / 100);
    std::uint8_t h = static_cast<std::uint8_t>(
        (q + ((13 * (m + 1U)) / 5U) + K + (K / 4) + (J / 4) + (5 * J)) % 7U
    );
    // Zeller: 0=Saturday, 1=Sunday, ..., 6=Friday
    // We want: 0=Sunday, 1=Monday, ..., 6=Saturday
    if(h == 0U)
    {
        return 6U;
    }
    else
    {
        return static_cast<std::uint8_t>(h - 1U);

}

constexpr std::array<const char*, 7U> weekday_names = {
    "Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"
};

std::uint8_t weekday_to_index(const std::string& weekday)
{
    for(std::uint8_t i = 0U; i < 7U; ++i)
    {
        if(weekday == weekday_names[i])
        {
            return i;
        }
    }
    throw std::invalid_argument("Invalid weekday name");
}

std::uint8_t days_in_month(meetup::year_t year, meetup::month_t month)
{
    // February leap year check
    if(month == 2U)
    {
        if(((year % 4) == 0) && (((year % 100) != 0) || ((year % 400) == 0)))
        {
            return 29U;
        }
        else
        {
            return 28U;
        }
    }
    // April, June, September, November
    if((month == 4U) || (month == 6U) || (month == 9U) || (month == 11U))
    {
        return 30U;
    }
    // All others
    return 31U;
}

    {
        m = static_cast<meetup::month_t>(m + 12U);
        y = static_cast<meetup::year_t>(y - 1);
    }
    meetup::day_t q = day;
    meetup::year_t K = static_cast<meetup::year_t>(y % 100);
    meetup::year_t J = static_cast<meetup::year_t>(y / 100);
    std::uint8_t h = static_cast<std::uint8_t>(
        (q + ((13 * (m + 1U)) / 5U) + K + (K / 4) + (J / 4) + (5 * J)) % 7U

    // We want: 0=Sunday, 1=Monday, ..., 6=Saturday
    if(h == 0U)
    {
        return 6U;
    }
    else
    {
        return static_cast<std::uint8_t>(h - 1U);
    }

meetup::day_t day(meetup::year_t year, meetup::month_t month, const std::string& weekday, const std::string& schedule)
{
    const std::uint8_t target_wd = weekday_to_index(weekday);
    const std::uint8_t days = days_in_month(year, month);

    if(schedule == "teenth")
    {
        for(meetup::day_t d = 13U; d <= 19U; ++d)
        {
            if(calc_weekday(year, month, d) == target_wd)
            {
                return d;
            }
        }
    }
    else if((schedule == "first") || (schedule == "second") || (schedule == "third") || (schedule == "fourth"))
    {
        std::uint8_t week = 0U;
        if(schedule == "first") { week = 1U; }
        else if(schedule == "second") { week = 2U; }
        else if(schedule == "third") { week = 3U; }
        else if(schedule == "fourth") { week = 4U; }

        meetup::day_t d = 1U;
        std::uint8_t found = 0U;
        for(; d <= days; ++d)
        {
            if(calc_weekday(year, month, d) == target_wd)
            {
                ++found;
                if(found == week)
                {
                    return d;
                }
            }
        }
    }
    else if(schedule == "last")
    {
        for(meetup::day_t d = days; d >= 1U; --d)
        {
            if(calc_weekday(year, month, d) == target_wd)
            {
                return d;
            }
        }
    }
    throw std::invalid_argument("Invalid schedule or no such day in month");
}
