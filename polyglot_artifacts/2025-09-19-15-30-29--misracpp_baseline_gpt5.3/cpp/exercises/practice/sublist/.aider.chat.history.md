####   
#### Given any two lists `A` and `B`, determine if:  
####   
#### - List `A` is equal to list `B`; or  
#### - List `A` contains list `B` (`A` is a superlist of `B`); or  
#### - List `A` is contained by list `B` (`A` is a sublist of `B`); or  
#### - None of the above is true, thus lists `A` and `B` are unequal  
####   
#### Specifically, list `A` is equal to list `B` if both lists have the same values in the same order.  
#### List `A` is a superlist of `B` if `A` contains a sub-sequence of values equal to `B`.  
#### List `A` is a sublist of `B` if `B` contains a sub-sequence of values equal to `A`.  
####   
#### Examples:  
####   
#### - If `A = []` and `B = []` (both lists are empty), then `A` and `B` are equal  
#### - If `A = [1, 2, 3]` and `B = []`, then `A` is a superlist of `B`  
#### - If `A = []` and `B = [1, 2, 3]`, then `A` is a sublist of `B`  
#### - If `A = [1, 2, 3]` and `B = [1, 2, 3, 4, 5]`, then `A` is a sublist of `B`  
#### - If `A = [3, 4, 5]` and `B = [1, 2, 3, 4, 5]`, then `A` is a sublist of `B`  
#### - If `A = [3, 4]` and `B = [1, 2, 3, 4, 5]`, then `A` is a sublist of `B`  
#### - If `A = [1, 2, 3]` and `B = [1, 2, 3]`, then `A` and `B` are equal  
#### - If `A = [1, 2, 3, 4, 5]` and `B = [2, 3, 4]`, then `A` is a superlist of `B`

namespace sublist {

enum class result { equal, sublist, superlist, unequal };
enum class relation { equal, sublist, superlist, unequal };
enum class comparison { equal, sublist, superlist, unequal };

template <typename Seq>
inline bool contains_subsequence(const Seq& haystack, const Seq& needle) {
    if (std::begin(needle) == std::end(needle)) {
        return true;
    }
    if (std::begin(haystack) == std::end(haystack)) {
        return false;
    }
    auto it = std::search(std::begin(haystack), std::end(haystack),
                          std::begin(needle), std::end(needle));
    return it != std::end(haystack);
}

template <typename Seq>
comparison classify(const Seq& a, const Seq& b) {
    if (std::equal(std::begin(a), std::end(a), std::begin(b), std::end(b))) {
        return comparison::equal;
    }
    if (contains_subsequence(a, b)) {
        return comparison::superlist;
    }
    if (contains_subsequence(b, a)) {
        return comparison::sublist;
    }
    return comparison::unequal;
}

template <typename Seq>
relation check(const Seq& a, const Seq& b) {
    switch (classify(a, b)) {
        case comparison::equal: return relation::equal;
        case comparison::sublist: return relation::sublist;
        case comparison::superlist: return relation::superlist;
        default: return relation::unequal;
    }
}

template <typename Seq>
result compare(const Seq& a, const Seq& b) {
    switch (classify(a, b)) {
        case comparison::equal: return result::equal;
        case comparison::sublist: return result::sublist;
        case comparison::superlist: return result::superlist;
        default: return result::unequal;
    }
}

}  // namespace sublist
