std::mutex                                 mtx_    {};

/* Character pools */
const std::string kLetters { "ABCDEFGHIJKLMNOPQRSTUVWXYZ" };

/* Random selection helpers */
std::string random_letters(std::uint32_t count)
{
    std::string result {};
    result.reserve(count);

    std::uniform_int_distribution<std::uint32_t> dist { static_cast<std::uint32_t>(0U),
                                                        static_cast<std::uint32_t>(kLetters.size() - 1U) };

    for (std::uint32_t i { 0U }; (i < count); ++i)
    {
        result.push_back(kLetters[dist(rng_)]);
    }

    return result;
}

std::string random_digits(std::uint32_t count)
{
    std::string result {};
    result.reserve(count);

    std::uniform_int_distribution<std::uint32_t> dist { static_cast<std::uint32_t>(0U),
                                                        static_cast<std::uint32_t>(kDigits.size() - 1U) };

    for (std::uint32_t i { 0U }; (i < count); ++i)
    {
        result.push_back(kDigits[dist(rng_)]);
    }

    return result;
}
}  // unnamed namespace

/* robot implementation
 * -------------------- */
robot::robot() : current_name_ {}
{
    /* Defer actual name creation until first call to name() */
}

const std::string& robot::name()
{
    if (current_name_.empty())
    {
        generate_name();
    }

    return current_name_;
}

void robot::reset()
{
    std::lock_guard<std::mutex> lock { mtx_ };

    if (!current_name_.empty())
    {
        (void)used_names_.erase(current_name_);
        current_name_.clear();
    }
}

void robot::generate_name()
{
    std::string candidate {};

    for (;;)
    {
        candidate = random_letters(static_cast<std::uint32_t>(2U)) +
                    random_digits(static_cast<std::uint32_t>(3U));

        /* Protect shared registry */
        std::lock_guard<std::mutex> lock { mtx_ };

        if (used_names_.find(candidate) == used_names_.end())
        {
            /* Unique – reserve and exit loop */
            (void)used_names_.insert(candidate);
            current_name_ = candidate;
            break;
        }
    }
}

}  // namespace robot_name
