void school::add(const std::string &name, int grade) {
    // First, ensure the student is not already recorded in any grade.
    for (auto &[g, students] : m_roster) {
        auto it = std::find(students.begin(), students.end(), name);
        if (it != students.end()) {
            if (g == grade) {
                // Already in the correct grade – nothing to do.
                return;
            }
            // Remove from the old grade to move them.
            students.erase(it);
            break;
        }
    }

    // Insert student into the requested grade.
    m_roster[grade].push_back(name);

    // Keep the vector sorted so all retrievals are already ordered.
    auto &students = m_roster[grade];
    std::sort(students.begin(), students.end());
}

school::roster_t school::roster() const {
    return m_roster; // std::map maintains grade order automatically
}

std::vector<std::string> school::grade(int grade) const {
    auto it = m_roster.find(grade);
    if (it == m_roster.end()) {
        return {};
    }
    return it->second;
}

}  // namespace grade_school
