auto solve(const std::vector<Item>& items, std::uint32_t capacity) -> Result
{
    const std::size_t item_count = items.size();

    std::vector<std::vector<std::uint32_t>> dp(
        item_count + 1U,
        std::vector<std::uint32_t>(capacity + 1U, 0U));

    std::vector<std::vector<bool>> keep(
        item_count + 1U,
        std::vector<bool>(capacity + 1U, false));

    for (std::size_t i = 1U; i <= item_count; ++i)
    {
        const std::uint32_t wt  = items[i - 1U].weight;
        const std::uint32_t val = items[i - 1U].value;

        for (std::uint32_t w = 0U; w <= capacity; ++w)
        {
            if ((wt <= w))
            {
                const std::uint32_t val_with    = val + dp[i - 1U][w - wt];
                const std::uint32_t val_without = dp[i - 1U][w];

                if ((val_with > val_without))
                {
                    dp[i][w]   = val_with;
                    keep[i][w] = true;
                }
                else
                {
                    dp[i][w] = val_without;
                }
            }
            else
            {
                dp[i][w] = dp[i - 1U][w];
            }
        }
    }

    std::vector<std::size_t> indices;
    indices.reserve(item_count);

    std::uint32_t w = capacity;
    for (std::size_t i = item_count; i > 0U; --i)
    {
        if (keep[i][w])
        {
            indices.push_back(i - 1U);
            w -= items[i - 1U].weight;
        }
    }

    std::reverse(indices.begin(), indices.end());

    const std::uint32_t total_value = dp[item_count][capacity];

    return Result{total_value, indices};
}

} // namespace knapsack
