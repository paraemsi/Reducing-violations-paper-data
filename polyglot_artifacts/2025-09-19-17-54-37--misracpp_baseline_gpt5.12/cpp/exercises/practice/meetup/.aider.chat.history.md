std::chrono::weekday to_chrono_weekday(weekday w) {
    using std::chrono::weekday;
    switch (w) {
        case weekday::Monday:    return weekday{1};
        case weekday::Tuesday:   return weekday{2};
        case weekday::Wednesday: return weekday{3};
        case weekday::Thursday:  return weekday{4};
        case weekday::Friday:    return weekday{5};

        case weekday::Sunday:    return weekday{0};
    }
    return weekday{0};
}
}  // namespace

scheduler::scheduler(int month, int year)
    : month_{month}, year_{year} {}

std::chrono::year_month_day scheduler::meetup_day(weekday day, schedule s) const {
    using namespace std::chrono;

    const year y{year_};
    const month m{static_cast<unsigned>(month_)};

    auto as_ymd = [&](int d) -> year_month_day {
        return year_month_day{y, m, day{static_cast<unsigned>(d)}};
    };

    const weekday target = to_chrono_weekday(day);

    if (s == schedule::teenth) {
        for (int d = 13; d <= 19; ++d) {
            sys_days sd{as_ymd(d)};
            if (weekday{sd} == target) {
                return year_month_day{sd};
            }
        }
        throw std::logic_error("No teenth day found");
    }

    // First day of the month
    sys_days first_sd{year_month_day{y, m, day{1}}};
    weekday first_wd{first_sd};
    days diff = target - first_wd;  // in range [0,6]
    int first_occurrence = 1 + static_cast<int>(diff.count());

    int d = first_occurrence;
    switch (s) {
        case schedule::first:
            break;
        case schedule::second:
            d += 7;
            break;
        case schedule::third:
            d += 14;
            break;
        case schedule::fourth:
            d += 21;
            break;
        case schedule::last: {
            year_month_day_last ymdl{y, month_day_last{m}};
            year_month_day last_ymd{ymdl};
            sys_days last_sd{last_ymd};
            weekday last_wd{last_sd};
            int back_days = static_cast<int>((last_wd - target).count());  // 0..6
            d = static_cast<int>(unsigned(last_ymd.day())) - back_days;
            break;
        }
        case schedule::teenth:  // handled above; keep to silence warnings
            break;
    }

    return as_ymd(d);
}

}  // namespace meetup

    const std::chrono::year y{year_};
    const std::chrono::month m{static_cast<unsigned>(month_)};

    auto as_ymd = [&](int d) -> std::chrono::year_month_day {
        return std::chrono::year_month_day{y, m, std::chrono::day{static_cast<unsigned>(d)}};

                return std::chrono::year_month_day{sd};
            }
        }

date scheduler::meetup(boost::date_time::weekdays wd, which w) const {
    const date first_day(year_, month_, 1);

    if (w == which::teenth) {
        for (int d = 13; d <= 19; ++d) {
            const date dt(year_, month_, d);
            if (dt.day_of_week().as_number() == static_cast<int>(wd)) {
                return dt;
            }
        }
        // Should never happen as each weekday occurs exactly once in the teenth range
        return first_day;
    }

    if (w == which::last) {
        const date last_day = first_day.end_of_month();
        const int back = (last_day.day_of_week().as_number() - static_cast<int>(wd) + 7) % 7;
        return last_day - date_duration(back);
    }

    // first..fourth
    const int target_index = static_cast<int>(wd);
    const int first_wd = first_day.day_of_week().as_number();
    const int offset = (target_index - first_wd + 7) % 7;
    const int n = static_cast<int>(w); // 1..4
    const int day = 1 + offset + 7 * (n - 1);
    return date(year_, month_, day);
}
