namespace meetup {

using namespace std::chrono;

static sys_days first_weekday_on_or_after(sys_days start, weekday wd) {
    weekday wstart{start};
    days diff = (wd - wstart);
    if (diff.count() < 0) diff += days{7};
    return start + diff;
}

std::chrono::year_month_day meetup_day(int year, unsigned month, std::chrono::weekday wd, schedule s) {
    if (month < 1 || month > 12) {
        throw std::out_of_range("month must be in 1..12");
    }

    const year y{year};
    const std::chrono::month m{month};

    sys_days first_of_month = sys_days(y/m/1);
    sys_days chosen;

    switch (s) {
        case schedule::first: {
            chosen = first_weekday_on_or_after(first_of_month, wd);
            break;
        }
        case schedule::second: {
            chosen = first_weekday_on_or_after(first_of_month, wd) + days{7};
            break;
        }
        case schedule::third: {
            chosen = first_weekday_on_or_after(first_of_month, wd) + days{14};
            break;
        }
        case schedule::fourth: {
            chosen = first_weekday_on_or_after(first_of_month, wd) + days{21};
            break;
        }
        case schedule::last: {
            sys_days last_of_month = sys_days(year_month_day_last{y/m/last});
            weekday wlast{last_of_month};
            int back = wlast.c_encoding() - wd.c_encoding();
            if (back < 0) back += 7;
            chosen = last_of_month - days{back};
            break;
        }
        case schedule::teenth: {
            sys_days first_occurrence = first_weekday_on_or_after(first_of_month, wd);
            year_month_day ymd_first{first_occurrence};
            unsigned d = unsigned(ymd_first.day());
            while (d < 13) {
                first_occurrence += days{7};
                d += 7;
            }
            if (d > 19) {
                first_occurrence -= days{7};
            }
            chosen = first_occurrence;
            break;
        }
    }

    return year_month_day{chosen};
}

}  // namespace meetup
