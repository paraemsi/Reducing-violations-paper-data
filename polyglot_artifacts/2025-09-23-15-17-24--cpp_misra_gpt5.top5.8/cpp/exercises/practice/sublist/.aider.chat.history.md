enum class result
{
    equal,

    superlist,
    unequal
};

template <typename T>
inline bool is_subsequence(const std::vector<T>& haystack, const std::vector<T>& needle)
{
    if (needle.empty())
    {
        return true;
    }
    else
    {
        const auto it = std::search(haystack.begin(), haystack.end(), needle.begin(), needle.end());
        return (it != haystack.end());
    }
}

template <typename T>
inline result check(const std::vector<T>& a, const std::vector<T>& b)
{
    if (a == b)
    {
        return result::equal;
    }

    const std::size_t a_size = a.size();
    const std::size_t b_size = b.size();

    if (a_size < b_size)
    {
        if (is_subsequence(b, a))
        {
            return result::sublist;
        }
        else
        {
            return result::unequal;
        }
    }
    else if (a_size > b_size)
    {
        if (is_subsequence(a, b))
        {
            return result::superlist;
        }
        else
        {
            return result::unequal;
        }
    }
    else
    {
        return result::unequal;
    }
}
