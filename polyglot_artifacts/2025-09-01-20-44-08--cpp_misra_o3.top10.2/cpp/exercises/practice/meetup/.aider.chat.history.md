scheduler::scheduler(std::int32_t year, std::uint32_t month) noexcept
    : m_year{std::chrono::year{year}},
      m_month{std::chrono::month{static_cast<unsigned>(month)}}
{
    // Precondition: month shall be in the range [1, 12]
    assert((month >= 1U) && (month <= 12U));
}

std::chrono::year_month_day
scheduler::get_day(std::chrono::weekday wd, schedule sc) const
{
    // Start from the first day of the month.
    const std::chrono::year_month_day first_dom{m_year, m_month, std::chrono::day{1}};
    const std::chrono::weekday         first_wd{first_dom};

    std::uint32_t day_num{1U};

    switch (sc)
    {
    case schedule::first:
    case schedule::second:
    case schedule::third:
    case schedule::fourth:
    {
        const std::uint32_t ordinal{static_cast<std::uint32_t>(sc)};
        const std::uint32_t delta{
            static_cast<std::uint32_t>((wd.iso_encoding() + 7 - first_wd.iso_encoding()) % 7)};
        day_num = (1U + delta) + (7U * (ordinal - 1U));
        break;
    }
    case schedule::teenth:
    {
        for (std::uint32_t d{13U}; d <= 19U; ++d)
        {
            const std::chrono::year_month_day ymd{m_year, m_month, std::chrono::day{d}};
            if (std::chrono::weekday{ymd} == wd)
            {
                day_num = d;
                break;
            }
        }
        break;
    }
    case schedule::last:
    {
        const std::chrono::year_month_day last_dom{
            std::chrono::year_month_day_last{m_year / m_month / std::chrono::last}};
        std::uint32_t d{to_uint(last_dom.day())};
        while (d > 0U)
        {
            const std::chrono::year_month_day ymd{m_year, m_month, std::chrono::day{d}};
            if (std::chrono::weekday{ymd} == wd)
            {
                day_num = d;
                break;
            }
            --d;
        }
        break;
    }
    default:
        // Unreachable, but silences some compilers
        assert(false);
    }

    return std::chrono::year_month_day{m_year, m_month, std::chrono::day{day_num}};
}
