std::vector<unsigned int> convert(const std::vector<unsigned int>& digits,
                                  unsigned int input_base,
                                  unsigned int output_base) {
    if (input_base < 2) {
        throw std::invalid_argument("input base must be >= 2");
    }
    if (output_base < 2) {
        throw std::invalid_argument("output base must be >= 2");
    }
    if (digits.empty()) {
        throw std::invalid_argument("digits must not be empty");
    }

    // Skip leading zeros
    std::size_t idx = 0;
    while (idx < digits.size() && digits[idx] == 0) {
        ++idx;
    }
    if (idx == digits.size()) {
        return {0};
    }

    unsigned long long value = 0;
    for (; idx < digits.size(); ++idx) {
        unsigned int d = digits[idx];
        if (d >= input_base) {
            throw std::invalid_argument("digit out of range for input base");
        }
        value = value * input_base + d;
    }

    std::vector<unsigned int> result;
    while (value > 0) {
        result.push_back(static_cast<unsigned int>(value % output_base));
        value /= output_base;
    }
    std::reverse(result.begin(), result.end());
    return result;
}
