static year_month_day compute_teenth(year y, month m, weekday wd) {
    for (int d = 13; d <= 19; ++d) {
        sys_days date = sys_days{y/m/day{d}};
        if (weekday{date} == wd) {
            return year_month_day{date};
        }
    }
    throw std::logic_error("No teenth day found for given weekday");
}

year_month_day meetup_day(year y, month m, weekday wd, schedule s) {
    switch (s) {
        case schedule::teenth:
            return compute_teenth(y, m, wd);
        case schedule::first:
        case schedule::second:
        case schedule::third:
        case schedule::fourth: {
            sys_days first = sys_days{y/m/day{1}};
            unsigned first_wd = weekday{first}.c_encoding(); // 0=Sun..6=Sat
            unsigned target = wd.c_encoding();
            unsigned diff = (target + 7u - first_wd) % 7u;
            unsigned occurrence = 0;
            switch (s) {
                case schedule::first: occurrence = 0; break;
                case schedule::second: occurrence = 1; break;
                case schedule::third: occurrence = 2; break;
                case schedule::fourth: occurrence = 3; break;
                default: break;
            }
            day d{static_cast<unsigned>(1 + diff + 7u * occurrence)};
            return year_month_day{y, m, d};
        }
        case schedule::last: {
            sys_days last_day = sys_days{y/m/std::chrono::last};
            unsigned last_wd = weekday{last_day}.c_encoding();
            unsigned target = wd.c_encoding();
            unsigned back = (last_wd + 7u - target) % 7u;
            sys_days date = last_day - days{static_cast<int>(back)};
            return year_month_day{date};
        }
        default:
            throw std::invalid_argument("Unknown schedule");
    }
}

    year y{year_i};
    month m{month_u};
    if (!m.ok()) {
        throw std::invalid_argument("month must be 1..12");
    }
    return meetup_day(y, m, wd, s);
}
