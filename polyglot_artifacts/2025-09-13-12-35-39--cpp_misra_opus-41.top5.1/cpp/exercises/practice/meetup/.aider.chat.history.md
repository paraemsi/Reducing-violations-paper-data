static std::uint32_t days_in_month(std::uint32_t year, std::uint32_t month) {
    if (month == 2U) {
        // February - check for leap year
        if ((((year % 4U) == 0U) && ((year % 100U) != 0U)) || ((year % 400U) == 0U)) {
            return 29U;
        }
        return 28U;
    }

        return 30U;
    }
    return 31U;
}

static std::uint32_t day_of_week(std::uint32_t year, std::uint32_t month, std::uint32_t day) {
    // Zeller's congruence algorithm for Gregorian calendar
    // Returns 1=Monday, 2=Tuesday, ..., 7=Sunday
    std::uint32_t m = month;
    std::uint32_t y = year;
    
    if (m < 3U) {
        m = m + 12U;
        y = y - 1U;
    }
    
    std::uint32_t k = y % 100U;
    std::uint32_t j = y / 100U;
    
    std::uint32_t h = (day + (((m + 1U) * 13U) / 5U) + k + (k / 4U) + (j / 4U) + (5U * j)) % 7U;
    
    // Convert from Zeller's output (0=Saturday, 1=Sunday, ..., 6=Friday)
    // to our format (1=Monday, ..., 7=Sunday)
    if (h == 0U) {
        return 6U; // Saturday
    }
    if (h == 1U) {
        return 7U; // Sunday
    }
    return h - 1U;
}
