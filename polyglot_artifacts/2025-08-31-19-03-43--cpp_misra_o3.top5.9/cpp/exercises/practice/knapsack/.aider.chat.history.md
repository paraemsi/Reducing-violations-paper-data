std::int32_t solveKnapsack(const std::vector<Item>& items, std::int32_t maxWeight)
{
    const std::size_t n{items.size()};
    std::vector<std::int32_t> dp(static_cast<std::size_t>(maxWeight + static_cast<std::int32_t>(1)), static_cast<std::int32_t>(0));

    for(std::size_t i{0U}; i < n; ++i)
    {
        const std::int32_t w{static_cast<std::int32_t>(items[i].weight)};
        const std::int32_t v{static_cast<std::int32_t>(items[i].value)};

        for(std::int32_t weight{maxWeight}; weight >= w; --weight)
        {
            const std::size_t idx{static_cast<std::size_t>(weight)};
            const std::size_t prevIdx{static_cast<std::size_t>(weight - w)};
            const std::int32_t candidate{v + dp[prevIdx]};
            if(candidate > dp[idx])
            {
                dp[idx] = candidate;
            }
        }
    }
    return dp[static_cast<std::size_t>(maxWeight)];
}

{

    {
        const std::int32_t w{static_cast<std::int32_t>(items[i - 1U].weight)};

        {
            const std::size_t idx{static_cast<std::size_t>(weight)};
            if(weight < w)
            {
                dp[i][idx] = dp[i - 1U][idx];

            else
            {
                const std::int32_t withoutItem{dp[i - 1U][idx]};
                const std::int32_t withItem{v + dp[i - 1U][static_cast<std::size_t>(weight - w)]};
                dp[i][idx] = (withItem > withoutItem) ? withItem : withoutItem;
            }
        }
    }

        }
    }
