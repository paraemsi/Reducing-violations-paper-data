// nth in [1..4]
std::chrono::year_month_day nth_weekday_of_month(std::chrono::year y,
                                                 std::chrono::month m,
                                                 std::chrono::weekday wd,
                                                 int n) {
    using namespace std::chrono;
    const sys_days first = sys_days{y / m / day{1}};
    const weekday wd_first{first};
    const days delta = (wd - wd_first + days{7}) % days{7};
    const sys_days date = first + delta + days{7} * (n - 1);
    return year_month_day{date};
}

std::chrono::year_month_day last_weekday_of_month(std::chrono::year y,
                                                  std::chrono::month m,
                                                  std::chrono::weekday wd) {
    using namespace std::chrono;
    const year_month_day_last ymdl{y, month_day_last{m}};
    const sys_days last_sd{ymdl};
    const weekday wd_last{last_sd};
    const days back = (wd_last - wd + days{7}) % days{7};
    const sys_days date = last_sd - back;
    return year_month_day{date};
}

std::chrono::year_month_day teenth_weekday_of_month(std::chrono::year y,
                                                    std::chrono::month m,
                                                    std::chrono::weekday wd) {
    using namespace std::chrono;
    const sys_days d13{y / m / day{13}};
    for (int i = 0; i < 7; ++i) {
        const sys_days cur = d13 + days{i};
        if (weekday{cur} == wd) {
            return year_month_day{cur};
        }
    }
    // Should never happen
    return year_month_day{d13};
}

} // namespace

std::chrono::year_month_day meetup_day(std::chrono::year y,
                                       std::chrono::month m,
                                       std::chrono::weekday wd,
                                       week schedule) {
    switch (schedule) {
        case week::first:  return nth_weekday_of_month(y, m, wd, 1);
        case week::second: return nth_weekday_of_month(y, m, wd, 2);
        case week::third:  return nth_weekday_of_month(y, m, wd, 3);
        case week::fourth: return nth_weekday_of_month(y, m, wd, 4);
        case week::last:   return last_weekday_of_month(y, m, wd);
        case week::teenth: return teenth_weekday_of_month(y, m, wd);
    }
    // Unreachable, but placate compilers that warn on missing return.
    return std::chrono::year_month_day{};
}

std::chrono::year_month_day meetup_day(int year,
                                       int month,
                                       std::chrono::weekday wd,
                                       week schedule) {
    return meetup_day(std::chrono::year{year},
                      std::chrono::month{static_cast<unsigned>(month)},
                      wd,
                      schedule);
}

}  // namespace meetup

                                                 std::chrono::month m,
                                                 std::chrono::weekday wd,
                                                 int n) {
    using namespace std::chrono;
    const sys_days first = sys_days{y / m / day{1}};
    const weekday wd_first{first};
    const days delta = (wd - wd_first + days{7}) % days{7};
    const sys_days date = first + delta + days{7} * (n - 1);
