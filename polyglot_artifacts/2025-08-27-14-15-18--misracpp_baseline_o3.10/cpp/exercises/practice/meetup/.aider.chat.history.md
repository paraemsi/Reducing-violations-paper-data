#include "meetup.h"

#include <stdexcept>

namespace meetup {

namespace {

// ------------------------------------------------------------------
// Helper utilities
// ------------------------------------------------------------------

// Leap-year check (Gregorian calendar rules)
constexpr bool is_leap(int year)
{
    return (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);
}

// Number of days in the given month of the given year.
constexpr unsigned int days_in_month(int year, unsigned int month)
{
    constexpr unsigned int table[]{31, 28, 31, 30, 31, 30,
                                   31, 31, 30, 31, 30, 31};

    if (month == 2)
        return is_leap(year) ? 29 : 28;

    return table[month - 1];
}

} // namespace (anon)

// ------------------------------------------------------------------
// scheduler implementation
// ------------------------------------------------------------------
scheduler::scheduler(int year, unsigned int month)
    : m_year{year}
    , m_month{month}
{
    if (month == 0 || month > 12)
        throw std::invalid_argument("month must be in the range 1‒12");
}

std::chrono::year_month_day
scheduler::get_day(const std::chrono::weekday& target_wd,
                   schedule                     sch) const
{
    using namespace std::chrono;

    // Weekday of the first day of the month
    const sys_days first_day = year{m_year}/month{m_month}/day{1};
    const weekday  wd_first{first_day};

    // Days to add from the 1st to reach the first occurrence of target_wd
    const unsigned offset =
        (static_cast<unsigned>(target_wd.c_encoding()) -
         static_cast<unsigned>(wd_first.c_encoding()) + 7) %
        7;

    unsigned day_number{0};

    switch (sch)
    {
    case schedule::first:
    case schedule::second:
    case schedule::third:
    case schedule::fourth: {
        const unsigned nth = static_cast<unsigned>(sch); // 0,1,2,3
        day_number         = 1 + offset + 7 * nth;
        break;
    }

    case schedule::teenth: {
        for (unsigned d = 13; d <= 19; ++d)
        {
            if (weekday{sys_days(year{m_year}/month{m_month}/day{d})} ==
                target_wd)
            {
                day_number = d;
                break;
            }
        }
        break;
    }

    case schedule::last: {
        const unsigned dim = days_in_month(m_year, m_month);
        const weekday  wd_last{sys_days(year{m_year}/month{m_month}/day{dim})};

        const unsigned diff =
            (static_cast<unsigned>(wd_last.c_encoding()) -
             static_cast<unsigned>(target_wd.c_encoding()) + 7) %
            7;

        day_number = dim - diff;
        break;
    }

    default:
        throw std::invalid_argument("unknown schedule value");
    }

    return year{m_year} / month{m_month} / day{day{day_number}};
}

}  // namespace meetup
