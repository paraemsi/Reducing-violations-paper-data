    // Helper: convert Weekday enum to tm_wday (0=Sunday, ..., 6=Saturday)
    int to_tm_wday(Weekday wd) {
        switch (wd) {
            case Weekday::Sunday:    return 0;
            case Weekday::Monday:    return 1;
            case Weekday::Tuesday:   return 2;
            case Weekday::Wednesday: return 3;
            case Weekday::Thursday:  return 4;
            case Weekday::Friday:    return 5;
            case Weekday::Saturday:  return 6;
        }
        return 0; // Should never reach here
    }

    // Helper: get number of days in a month
    int days_in_month(int year, int month) {
        static const int days[] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
        if (month < 1 || month > 12) return 0;
        if (month == 2) { // February, check for leap year
            if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0))
                return 29;
        }
        return days[month-1];
    }
}

int day(int year, int month, Weekday weekday, Week week) {
    int target_wday = to_tm_wday(weekday);

    if (week == Week::teenth) {
        // Search 13th to 19th
        for (int d = 13; d <= 19; ++d) {
            std::tm t = {};
            t.tm_year = year - 1900;
            t.tm_mon = month - 1;
            t.tm_mday = d;
            std::mktime(&t);
            if (t.tm_wday == target_wday) {
                return d;
            }
        }
        throw std::logic_error("No teenth weekday found");
    }

    {
        std::tm t = {};

        t.tm_mon = month - 1;
        t.tm_mday = 1;
        std::mktime(&t);
        first_day_wday = t.tm_wday;
    }

    int day_of_month = 1 + ((target_wday - first_day_wday + 7) % 7);

    if (week == Week::first) {
        // already set
    } else if (week == Week::second) {
        day_of_month += 7;
    } else if (week == Week::third) {
        day_of_month += 14;

        day_of_month += 21;

        for (int d = dim; d >= dim - 6; --d) {

            t.tm_mday = d;

            if (t.tm_wday == target_wday) {
                return d;

        throw std::logic_error("No last weekday found");

        throw std::logic_error("Unknown week value");

    int dim = days_in_month(year, month);

                return 29;
