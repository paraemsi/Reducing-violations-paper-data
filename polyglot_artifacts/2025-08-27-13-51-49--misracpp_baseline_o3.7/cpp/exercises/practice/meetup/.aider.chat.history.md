    using namespace std::chrono;
    const year yr{year};
    const month mon{month};

std::chrono::year_month_day meetup_day(int year, unsigned month,
                                       std::chrono::weekday weekday,
                                       schedule sched) {
    using namespace std::chrono;

    const year yr{year};
    const month mon{month};

    if (!mon.ok()) {
        throw std::invalid_argument("month must be 1..12");
    }

    auto compute_from_first = [&](unsigned index) -> year_month_day {
        sys_days first = sys_days{yr / mon / day{1}};
        weekday first_wd{first};
        unsigned offset = static_cast<unsigned>((weekday - first_wd).count() + 7) % 7;
        unsigned d = 1 + offset + index * 7;
        if (d > days_in_month(year, month)) {
            throw std::domain_error("Requested occurrence does not exist in this month");
        }
        return year_month_day{yr, mon, day{d}};
    };

    switch (sched) {
        case schedule::teenth: {
            for (unsigned d = 13; d <= 19; ++d) {
                if (weekday{sys_days{yr / mon / day{d}}} == weekday) {
                    return year_month_day{yr, mon, day{d}};
                }
            }
            break; // should not reach
        }
        case schedule::first:
            return compute_from_first(0);
        case schedule::second:
            return compute_from_first(1);
        case schedule::third:
            return compute_from_first(2);
        case schedule::fourth:
            return compute_from_first(3);
        case schedule::fifth:
            return compute_from_first(4);
        case schedule::last: {
            year_month_day last_ymd = yr / mon / std::chrono::last;
            sys_days last_sd{last_ymd};
            weekday last_wd{last_sd};
            unsigned offset = static_cast<unsigned>((last_wd - weekday).count() + 7) % 7;
            unsigned d = static_cast<unsigned>(last_ymd.day()) - offset;
            return year_month_day{yr, mon, day{d}};
        }
    }
    throw std::logic_error("Unable to compute meetup day");
}

scheduler::scheduler(int year, unsigned month)
    : m_year(year), m_month(month) {}

std::chrono::year_month_day scheduler::get_day(std::chrono::weekday weekday,
                                               schedule sched) const {
    return meetup_day(m_year, m_month, weekday, sched);
}
