    int target_wday;
    if (weekday == "Sunday") target_wday = 0;
    else if (weekday == "Monday") target_wday = 1;
    else if (weekday == "Tuesday") target_wday = 2;
    else if (weekday == "Wednesday") target_wday = 3;
    else if (weekday == "Thursday") target_wday = 4;
    else if (weekday == "Friday") target_wday = 5;

    time_in.tm_isdst = -1;

    std::mktime(&time_in);

        for (int day = 13; day <= 19; ++day) {

            temp.tm_mday = day;

        int day_delta = (target_wday - time_in.tm_wday + 7) % 7;
