    if (m < 1 || m > 12) {
        throw std::out_of_range("Month must be in [1, 12]");

inline int days_in_month(int y, unsigned m) {
    static const int days[] = {31,28,31,30,31,30,31,31,30,31,30,31};
    if (m == 2) {
        return days[1] + (is_leap(y) ? 1 : 0);

    return days[m - 1];

inline int weekday_to_num(weekday wd) {
    switch (wd) {
        case weekday::sunday:    return 0;
        case weekday::monday:    return 1;
        case weekday::tuesday:   return 2;
        case weekday::wednesday: return 3;
        case weekday::thursday:  return 4;
        case weekday::friday:    return 5;
        case weekday::saturday:  return 6;
    }
    return 0;
}

inline bool is_leap(int y) {
    return (y % 4 == 0 && y % 100 != 0) || (y % 400 == 0);
}

inline int days_in_month(int y, unsigned m) {
    static const int days[] = {31,28,31,30,31,30,31,31,30,31,30,31};
    if (m == 2) {
        return days[1] + (is_leap(y) ? 1 : 0);
    }
    return days[m - 1];
}

// Sakamoto's algorithm: returns 0=Sunday..6=Saturday
inline int weekday_for_date(int y, unsigned m, unsigned d) {
    static const int t[] = {0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4};
    if (m < 3) y -= 1;
    return (y + y/4 - y/100 + y/400 + t[m - 1] + static_cast<int>(d)) % 7;
}

    if (m < 1 || m > 12) {
        throw std::out_of_range("Month must be in [1, 12]");
    }

    switch (sc) {

            if (delta < 0) {
