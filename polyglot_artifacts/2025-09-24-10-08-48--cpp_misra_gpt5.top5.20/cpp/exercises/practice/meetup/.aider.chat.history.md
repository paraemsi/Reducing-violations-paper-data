#include <chrono>
#include <cstdint>
#include <stdexcept>

namespace meetup {

namespace {

// Compute the nth (1-4) occurrence of a weekday in a given year/month.
static std::chrono::year_month_day nth_weekday_of_month(std::chrono::year y,
                                                        std::chrono::month m,
                                                        std::chrono::weekday wd,
                                                        std::uint32_t n)
{
    // n is 1..4 by contract
    const std::chrono::year_month_weekday ymwd{
        y, m, std::chrono::weekday_indexed{wd, static_cast<unsigned int>(n)}
    };
    const std::chrono::sys_days sd{ymwd};
    return std::chrono::year_month_day{sd};
}

// Compute the last occurrence of a weekday in a given year/month.
static std::chrono::year_month_day last_weekday_of_month(std::chrono::year y,
                                                         std::chrono::month m,
                                                         std::chrono::weekday wd)
{
    const std::chrono::year_month_weekday ymwd{
        y, m, std::chrono::weekday_last{wd}
    };
    const std::chrono::sys_days sd{ymwd};
    return std::chrono::year_month_day{sd};
}

} // anonymous namespace

scheduler::scheduler(std::int32_t year, std::uint32_t month) noexcept
    : year_{std::chrono::year{year}}
    , month_{std::chrono::month{static_cast<unsigned int>(month)}}
{
    // No body needed; basic member initialization only.
}

std::chrono::year_month_day scheduler::day(std::chrono::weekday wd, schedule s) const
{
    switch (s) {
        case schedule::first: {
            return nth_weekday_of_month(year_, month_, wd, static_cast<std::uint32_t>(1U));
        }
        case schedule::second: {
            return nth_weekday_of_month(year_, month_, wd, static_cast<std::uint32_t>(2U));
        }
        case schedule::third: {
            return nth_weekday_of_month(year_, month_, wd, static_cast<std::uint32_t>(3U));
        }
        case schedule::fourth: {
            return nth_weekday_of_month(year_, month_, wd, static_cast<std::uint32_t>(4U));
        }
        case schedule::last: {
            return last_weekday_of_month(year_, month_, wd);
        }
        case schedule::teenth: {
            // Search the teenth days 13..19 for the matching weekday.
            for (std::int32_t d = static_cast<std::int32_t>(13); (d <= static_cast<std::int32_t>(19)); ++d) {
                const std::chrono::year_month_day ymd{
                    year_, month_, std::chrono::day{static_cast<unsigned int>(d)}
                };
                const std::chrono::weekday cur_wd{std::chrono::sys_days{ymd}};
                if ((cur_wd == wd)) {
                    return ymd;
                }
            }
            // Should never happen for valid calendar data.
            throw std::logic_error("No teenth weekday found");
        }
        default: {
            throw std::invalid_argument("Unknown schedule");
        }
    }
}

}  // namespace meetup
