namespace phone_number {

auto clean_number(const std::string& input) -> std::string
{
    std::string digits{};
    digits.reserve(input.size());

    for (char ch : input) {
        if ((std::isdigit(std::char_traits<char>::to_int_type(ch)) != 0)) {
            digits.push_back(ch);
        }
    }

    if ((digits.size() == static_cast<std::string::size_type>(11U)) && (digits.front() == '1')) {
        digits.erase(digits.begin());
    }

    return digits;
}
