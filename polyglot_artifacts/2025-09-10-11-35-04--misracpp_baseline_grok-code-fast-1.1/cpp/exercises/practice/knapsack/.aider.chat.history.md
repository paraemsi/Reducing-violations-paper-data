#include "knapsack.h"
#include <vector>
#include <utility>
#include <algorithm>

namespace knapsack
{

std::pair<int, std::vector<int>> solveKnapsack(const std::vector<Item>& items, int capacity)
{
    int n = items.size();
    std::vector<std::vector<int>> dp(n + 1, std::vector<int>(capacity + 1, 0));

    // Fill the DP table
    for (int i = 1; i <= n; ++i)
    {
        for (int w = 0; w <= capacity; ++w)
        {
            if (items[i - 1].weight <= w)
            {
                dp[i][w] = std::max(dp[i - 1][w], dp[i - 1][w - items[i - 1].weight] + items[i - 1].value);
            }
            else
            {
                dp[i][w] = dp[i - 1][w];
            }
        }
    }

    int max_value = dp[n][capacity];

    // Backtrack to find selected items
    std::vector<int> selected;
    int w = capacity;
    for (int i = n; i > 0; --i)
    {
        if (dp[i][w] != dp[i - 1][w])
        {
            selected.push_back(i - 1); // Store the index
            w -= items[i - 1].weight;
        }
    }

    // Reverse to get indices in ascending order
    std::reverse(selected.begin(), selected.end());

    return {max_value, selected};
}

} // namespace knapsack
