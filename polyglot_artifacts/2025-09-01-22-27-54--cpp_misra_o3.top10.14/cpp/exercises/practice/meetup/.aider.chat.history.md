                            const std::uint8_t                    nth) const
    -> boost::gregorian::date

    /*  first day of given month */
    const boost::gregorian::date first_day {year_, static_cast<unsigned>(month_), 1U};

    /*  distance (0-6) to requested weekday */
    const std::uint8_t diff {static_cast<std::uint8_t>(
        ( (static_cast<std::uint8_t>(wd.as_number()) +
           (7U - static_cast<std::uint8_t>(first_day.day_of_week()))) % 7U ))};

    -> boost::gregorian::date

    /*  get first day of next month then step back */

                               ? boost::gregorian::Jan

                                     static_cast<unsigned>(month_) + 1U)};
