std::chrono::year_month_day scheduler::get_day(std::chrono::weekday wd, schedule sc) const {
    auto make_ymd = [this](unsigned int d) {
        return std::chrono::year_month_day{year_, month_, std::chrono::day{d}};
    };

    switch (sc) {
        case schedule::teenth: {
            for (unsigned int d = 13; d <= 19; ++d) {
                auto ymd = make_ymd(d);
                if (std::chrono::weekday{ymd} == wd) {
                    return ymd;
                }
            }
            break;
        }
        case schedule::last: {
            std::chrono::year_month_day_last ymdl{year_, month_, std::chrono::last};
            int last_dom = static_cast<int>(static_cast<unsigned>(ymdl.day()));
            for (int d = last_dom; d >= last_dom - 6; --d) {
                auto ymd = make_ymd(static_cast<unsigned>(d));
                if (std::chrono::weekday{ymd} == wd) {
                    return ymd;
                }
            }
            break;
        }
        default: {
            int week_index = 0;
            switch (sc) {
                case schedule::first:  week_index = 0; break;
                case schedule::second: week_index = 1; break;
                case schedule::third:  week_index = 2; break;
                case schedule::fourth: week_index = 3; break;
                default: break;
            }
            unsigned int start_day = 1u + static_cast<unsigned>(week_index) * 7u;
            unsigned int end_day   = start_day + 6u;

                if (ymd.ok() && std::chrono::weekday{ymd} == wd) {
                    return ymd;
                }
            }
            break;
        }
    }

}
