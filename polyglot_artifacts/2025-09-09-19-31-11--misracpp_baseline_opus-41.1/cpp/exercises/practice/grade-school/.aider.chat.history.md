```cpp
#if !defined(GRADE_SCHOOL_H)
#define GRADE_SCHOOL_H

#include <map>
#include <set>
#include <string>
#include <vector>

namespace grade_school {

class school {
public:
    // Add a student to a grade
    // Returns false if student already exists in any grade
    bool add(const std::string& student_name, int grade);
    
    // Get all students in a specific grade, sorted alphabetically
    std::vector<std::string> grade(int grade_number) const;
    
    // Get all students in all grades
    // Returns a map where keys are grades (sorted) and values are sorted student names
    std::map<int, std::vector<std::string>> roster() const;

private:
    // Store students by grade, using set for automatic alphabetical sorting
    std::map<int, std::set<std::string>> roster_;
    
    // Track all students to prevent duplicates across grades
    std::set<std::string> all_students_;
};

}  // namespace grade_school

#endif // GRADE_SCHOOL_H
```

grade_school.cpp
```cpp
#include "grade_school.h"
#include <algorithm>

namespace grade_school {

bool school::add(const std::string& student_name, int grade) {
    // Check if student already exists in any grade
    if (all_students_.find(student_name) != all_students_.end()) {
        return false;  // Student already exists
    }
    
    // Add student to the specified grade
    roster_[grade].insert(student_name);
    
    // Track student in all_students set
    all_students_.insert(student_name);
    
    return true;
}

std::vector<std::string> school::grade(int grade_number) const {
    // Check if the grade exists in the roster
    auto it = roster_.find(grade_number);
    if (it != roster_.end()) {
        // Convert set to vector (already sorted alphabetically)
        return std::vector<std::string>(it->second.begin(), it->second.end());
    }
    
    // Return empty vector if grade doesn't exist
    return std::vector<std::string>();
}

std::map<int, std::vector<std::string>> school::roster() const {
    std::map<int, std::vector<std::string>> result;
    
    // Convert each grade's set of students to a vector
    for (const auto& grade_pair : roster_) {
        result[grade_pair.first] = std::vector<std::string>(
            grade_pair.second.begin(), 
            grade_pair.second.end()
        );
    }
    
    return result;
}

}  // namespace grade_school
```

2. **`grade` method**: Returns a sorted list of students in a specific grade
3. **`roster` method**: Returns the complete roster with grades sorted numerically and students sorted alphabetically within each grade

The solution uses:
- `std::map` to automatically sort grades numerically
- `std::set` to automatically sort student names alphabetically and ensure uniqueness within a grade
- A separate `std::set` to track all students across all grades to prevent duplicates
