        }

phone_number::phone_number(const std::string& input) : digits_() {
    std::string digits = extract_digits(input);

    // Handle country code '1'
    if ((digits.size() == 11U) && (digits[0] == '1')) {
        digits = digits.substr(1U);
    } else if (digits.size() == 11U) {
        throw std::invalid_argument("invalid country code");
    }

    if (digits.size() != 10U) {
        throw std::invalid_argument("invalid number of digits");

        throw std::invalid_argument("invalid area code");
    }
    if (!is_n_digit_char(digits[3])) {
        throw std::invalid_argument("invalid exchange code");
    }

    digits_ = digits;
}

    return digits_;
}

std::string phone_number::area_code() const {
