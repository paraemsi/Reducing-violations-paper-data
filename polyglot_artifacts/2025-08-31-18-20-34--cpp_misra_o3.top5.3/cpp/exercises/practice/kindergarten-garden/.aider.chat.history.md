garden::garden(const std::string& diagram, std::vector<std::string> students)
{
    if(students.empty())
    {
        m_students = {"Alice", "Bob", "Charlie", "David", "Eve", "Fred",
                      "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"};
    }
    else
    {
        m_students = students;
        std::sort(m_students.begin(), m_students.end());
    }

    std::istringstream iss(diagram);
    (void)std::getline(iss, m_row1);
    (void)std::getline(iss, m_row2);
}

std::vector<plant> garden::plants(const std::string& student) const
{
    auto it = std::find(m_students.begin(), m_students.end(), student);
    if(it == m_students.end())
    {
        return {};
    }

    std::uint8_t index = static_cast<std::uint8_t>(std::distance(m_students.begin(), it));
    std::size_t col_start = static_cast<std::size_t>(index) * static_cast<std::size_t>(2U);

    std::vector<plant> result;
    result.reserve(4U);

    result.push_back(char_to_plant(m_row1[col_start]));
    result.push_back(char_to_plant(m_row1[col_start + static_cast<std::size_t>(1U)]));
    result.push_back(char_to_plant(m_row2[col_start]));
    result.push_back(char_to_plant(m_row2[col_start + static_cast<std::size_t>(1U)]));

    return result;
}

plant garden::char_to_plant(char symbol)
{
    switch(symbol)
    {
        case 'G':
            return plant::grass;
        case 'C':
            return plant::clover;
        case 'R':
            return plant::radishes;
        case 'V':
            return plant::violets;
        default:
            /* Invalid symbol - returning grass as default */
            return plant::grass;
    }
}
