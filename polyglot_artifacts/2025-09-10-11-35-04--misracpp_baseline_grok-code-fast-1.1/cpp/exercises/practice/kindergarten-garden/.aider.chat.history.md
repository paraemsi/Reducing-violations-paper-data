#include "kindergarten_garden.h"
#include <algorithm>
#include <stdexcept>

namespace kindergarten_garden {

KindergartenGarden::KindergartenGarden(const std::vector<std::string>& diagram)
    : garden_(diagram),
      children_({"Alice", "Bob", "Charlie", "David", "Eve", "Fred", "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"}),
      plant_names_({{'G', "grass"}, {'C', "clover"}, {'R', "radishes"}, {'V', "violets"}}) {
    if (garden_.size() != 2 || garden_[0].size() != 12 || garden_[1].size() != 12) {
        throw std::invalid_argument("Invalid garden diagram");
    }
}

std::vector<std::string> KindergartenGarden::plants(const std::string& child) const {
    auto it = std::find(children_.begin(), children_.end(), child);
    if (it == children_.end()) {
        throw std::invalid_argument("Unknown child");
    }
    int index = it - children_.begin();
    std::vector<std::string> result;
    result.push_back(plant_names_.at(garden_[0][2 * index]));
    result.push_back(plant_names_.at(garden_[0][2 * index + 1]));
    result.push_back(plant_names_.at(garden_[1][2 * index]));
    result.push_back(plant_names_.at(garden_[1][2 * index + 1]));
    return result;
}

}  // namespace kindergarten_garden

    std::getline(iss, row1);
    std::getline(iss, row2);
