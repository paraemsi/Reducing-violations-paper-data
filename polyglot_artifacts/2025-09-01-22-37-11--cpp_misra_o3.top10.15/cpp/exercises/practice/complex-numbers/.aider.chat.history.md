Complex operator-(const Complex& lhs, const Complex& rhs) noexcept
{
    return Complex{(lhs.real - rhs.real), (lhs.imag - rhs.imag)};
}

Complex operator*(const Complex& lhs, const Complex& rhs) noexcept
{
    return Complex{((lhs.real * rhs.real) - (lhs.imag * rhs.imag)),
                   ((lhs.imag * rhs.real) + (lhs.real * rhs.imag))};
}

Complex operator/(const Complex& lhs, const Complex& rhs)
{
    const double denominator{((rhs.real * rhs.real) + (rhs.imag * rhs.imag))};

    if (denominator == 0.0)
    {
        throw std::domain_error("Division by zero complex number");
    }

    return Complex{(((lhs.real * rhs.real) + (lhs.imag * rhs.imag)) /
                    denominator),
                   (((lhs.imag * rhs.real) - (lhs.real * rhs.imag)) /
                    denominator)};
}

Complex conjugate(const Complex& value) noexcept
{
    return Complex{value.real, (-value.imag)};
}

double abs(const Complex& value) noexcept
{
    return std::sqrt(((value.real * value.real) +
                      (value.imag * value.imag)));
}

Complex exp(const Complex& value)
{
    const double expReal{std::exp(value.real)};
    const double cosImag{std::cos(value.imag)};
    const double sinImag{std::sin(value.imag)};

    return Complex{(expReal * cosImag), (expReal * sinImag)};
}
