date scheduler::find_date(int week, greg_weekday weekday) const {
    // Start from the first day of the month
    date first_day(year_, month_, 1);
    // Find the first occurrence of the target weekday
    date first_occurrence = first_day.end_of_day() + days((weekday - first_day.day_of_week()).as_number() % 7);
    if (first_occurrence < first_day) {
        first_occurrence += weeks(1);

    // Add appropriate number of weeks
