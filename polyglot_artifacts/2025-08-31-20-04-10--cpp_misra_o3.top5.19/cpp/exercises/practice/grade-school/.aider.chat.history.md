void school::add(const std::string& name, std::int32_t grade) {
    for (auto& entry : roster_) {
        const auto& students = entry.second;
        if ((std::find(students.begin(), students.end(), name)) != students.end()) {
            /* Student already enrolled – ignore duplicate */
            return;
        }
    }

    auto& students_in_grade = roster_[grade];
    students_in_grade.push_back(name);

    /* Keep students in alphabetical order */
    std::sort(students_in_grade.begin(), students_in_grade.end());
}

std::map<std::int32_t, std::vector<std::string>> school::roster() const {
    return roster_;
}

std::vector<std::string> school::grade(std::int32_t grade) const {
    const auto it = roster_.find(grade);
    if (it != roster_.end()) {
        return it->second;
    } else {
        return {};
    }
}
