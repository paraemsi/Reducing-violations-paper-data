// Helper function to get the day of the week (0=Sunday, 6=Saturday)
std::int32_t get_day_of_week(std::int32_t y, std::int32_t m, std::int32_t d) {
    std::int32_t temp_m = m;
    std::int32_t temp_y = y;
    if (temp_m < 3) {
        temp_m += 12;
        temp_y -= 1;
    }
    std::int32_t k = temp_y % 100;
    std::int32_t j = temp_y / 100;
    // Use parentheses to ensure proper precedence
    std::int32_t h = (d + ((13 * (temp_m + 1)) / 5) + k + (k / 4) + (j / 4) + (5 * j)) % 7;
    // Convert to 0=Sunday, 6=Saturday
    return ((h + 5) % 7);
}

    std::int32_t days_in_month = 0;
