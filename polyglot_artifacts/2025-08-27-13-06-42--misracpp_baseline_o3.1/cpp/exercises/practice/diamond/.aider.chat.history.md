namespace diamond {

std::string make(char letter) {
    if (letter < 'A' || letter > 'Z') {
        throw std::invalid_argument("letter must be between 'A' and 'Z'");
    }

    const int radius = letter - 'A';           // zero-based index of widest row
    const int line_length = radius * 2 + 1;    // width == height of the square
    std::string result;
    result.reserve((line_length + 1) * (radius * 2 + 1));  // rough estimate

    auto build_line = [&](int row) -> std::string {
        const char ch = static_cast<char>('A' + row);
        const int padding = radius - row;              // leading / trailing
        const int inner_spaces = row == 0 ? 0 : row * 2 - 1;

        std::string line;
        line.reserve(line_length);
        line.append(padding, ' ');
        line.push_back(ch);
        if (inner_spaces) {
            line.append(inner_spaces, ' ');
            line.push_back(ch);
        }
        line.append(padding, ' ');
        return line;
    };

    // Top half including the middle row
    for (int row = 0; row <= radius; ++row) {
        result += build_line(row);
        result.push_back('\n');
    }
    // Bottom half (exclude the middle row to avoid duplication)
    for (int row = radius - 1; row >= 0; --row) {
        result += build_line(row);
        if (row != 0) {            // no trailing newline on very last line
            result.push_back('\n');
        }
    }
    return result;
}

}  // namespace diamond
