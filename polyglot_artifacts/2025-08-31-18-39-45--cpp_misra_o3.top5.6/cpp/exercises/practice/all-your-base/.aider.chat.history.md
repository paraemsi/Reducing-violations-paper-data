        {
            throw std::invalid_argument("Digit out of range for input base");
        }

    /* Handle zero value */

    {
        if (d != 0)
        {

            break;
        }

    /* Convert to an unsigned integer value */

        value = value / static_cast<std::uint64_t>(to_base);
