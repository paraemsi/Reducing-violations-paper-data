static std::chrono::weekday to_chrono_weekday(weekday w) {
    using std::chrono::weekday;
    switch (w) {
        case weekday::Monday:    return weekday{1};
        case weekday::Tuesday:   return weekday{2};
        case weekday::Wednesday: return weekday{3};
        case weekday::Thursday:  return weekday{4};
        case weekday::Friday:    return weekday{5};
        case weekday::Saturday:  return weekday{6};
        case weekday::Sunday:    return weekday{0};
    }
    return weekday{0}; // Unreachable
}

std::chrono::year_month_day meetup_day(int year, int month, weekday wd, schedule which) {
    using namespace std::chrono;

    if (month < 1 || month > 12) {
        throw std::out_of_range("month must be in 1..12");
    }

    const year y{year};
    const std::chrono::month m{static_cast<unsigned>(month)};
    const weekday desired = to_chrono_weekday(wd);

    if (which == schedule::teenth) {
        for (unsigned d = 13; d <= 19; ++d) {
            sys_days sd = sys_days{year_month_day{y, m, day{d}}};
            if (weekday{sd} == desired) {
                return year_month_day{y, m, day{d}};
            }
        }
        throw std::runtime_error("No teenth day found");
    }

    // First day of the month
    sys_days first = sys_days{year_month_day{y, m, day{1}}};
    int fi = weekday{first}.c_encoding(); // 0=Sun .. 6=Sat
    int di = desired.c_encoding();

    unsigned start_day = 1u + static_cast<unsigned>((di - fi + 7) % 7);

    unsigned target_day = start_day;
    switch (which) {
        case schedule::first:  target_day = start_day; break;
        case schedule::second: target_day = start_day + 7; break;
        case schedule::third:  target_day = start_day + 14; break;
        case schedule::fourth: target_day = start_day + 21; break;
        case schedule::last: {
            year_month_day_last ymdl{y, month_day_last{m}};
            sys_days last_sd{ymdl};
            int li = weekday{last_sd}.c_encoding();
            unsigned last_dom = static_cast<unsigned>(year_month_day{last_sd}.day());
            unsigned back = static_cast<unsigned>((li - di + 7) % 7);
            target_day = last_dom - back;
            break;
        }
        case schedule::teenth:
            // already handled above
            break;
    }

    return year_month_day{y, m, day{target_day}};
}

}  // namespace meetup
