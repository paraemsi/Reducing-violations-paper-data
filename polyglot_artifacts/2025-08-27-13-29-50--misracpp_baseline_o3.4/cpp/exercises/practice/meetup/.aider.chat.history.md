scheduler::scheduler(int month, int year)
    : m_month(month), m_year(year) {
    if (month < 1 || month > 12) {
        throw std::invalid_argument{"month must be in 1..12"};
    }
}

std::chrono::year_month_day scheduler::day(std::chrono::weekday weekday,
                                           schedule s) const {
    using namespace std::chrono;

    const year y{m_year};
    const month m{static_cast<unsigned>(m_month)};

    auto first_of_month = year_month_day{y, m, day{1}};
    auto first_wd       = weekday{first_of_month};

    // Helper to obtain the last day number of the month
    const auto last_day = static_cast<unsigned>(
        year_month_day_last{y, month_day_last{m}}.day());

    unsigned day_number{1}; // will be adjusted below

    switch (s) {
    case schedule::first:
        day_number = 1;
        break;
    case schedule::second:
        day_number = 8;
        break;
    case schedule::third:
        day_number = 15;
        break;
    case schedule::fourth:
        day_number = 22;
        break;
    case schedule::fifth:
        day_number = 29;
        break;
    case schedule::teenth:
        day_number = 13;
        break;
    case schedule::last:
        day_number = last_day;
        break;
    }

    if (s == schedule::last) {
        // Walk *backwards* from the last day to the requested weekday
        weekday last_wd{year_month_day{y, m, day{day_number}}};
        unsigned diff = (static_cast<unsigned>(last_wd) + 7u -
                         static_cast<unsigned>(weekday)) %
                        7u;
        day_number -= diff;
    } else {
        // Walk *forwards* from the anchor day to the requested weekday
        weekday anchor_wd{year_month_day{y, m, day{day_number}}};
        unsigned diff = (static_cast<unsigned>(weekday) + 7u -
                         static_cast<unsigned>(anchor_wd)) %
                        7u;
        day_number += diff;

        if (day_number > last_day) {
            throw std::domain_error{"requested occurrence does not exist"};
        }
    }

    return year_month_day{y, m, day{day_number}};
}

} // namespace meetup

date scheduler::day(boost::date_time::weekdays desired_weekday,
                    schedule s) const {
    const int y = m_year;
    const int m = m_month;

    // Handle "teenth" specially
    if (s == schedule::teenth) {
        for (int d = 13; d <= 19; ++d) {
            date current{y, m, d};
            if (current.day_of_week().as_number() == desired_weekday) {
                return current;
            }
        }
        throw std::logic_error{"no teenth day found – unreachable"};
    }

    // Handle "last" by walking backwards from end-of-month
    if (s == schedule::last) {
        int last_dom = boost::gregorian::gregorian_calendar::end_of_month_day(
            y, static_cast<unsigned short>(m));
        for (int d = last_dom; d >= last_dom - 6; --d) {
            date current{y, m, d};
            if (current.day_of_week().as_number() == desired_weekday) {
                return current;
            }
        }
        throw std::logic_error{"no last weekday found – unreachable"};
    }

    // Ordinal occurrences (1-5)
    int index = ordinal_index(s);
    int day_start = 1 + 7 * (index - 1);          // 1,8,15,22,29
    for (int d = day_start; d < day_start + 7; ++d) {
        // Guard against months that have no 5th occurrence
        if (d > boost::gregorian::gregorian_calendar::end_of_month_day(
                    y, static_cast<unsigned short>(m))) {
            throw std::domain_error{"requested occurrence does not exist"};
        }
        date current{y, m, d};
        if (current.day_of_week().as_number() == desired_weekday) {
            return current;
        }
    }

    throw std::logic_error{"weekday not found – unreachable"};
}
