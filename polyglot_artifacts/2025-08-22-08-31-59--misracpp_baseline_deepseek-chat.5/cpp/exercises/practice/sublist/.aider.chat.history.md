        return ListComparison::equal;
    }
    
    // Check if list1 is a sublist of list2
    // A sublist means list1 appears as a contiguous sequence in list2
    if (list1.size() <= list2.size()) {
        for (size_t i = 0; i <= list2.size() - list1.size(); ++i) {
            bool found = true;
            for (size_t j = 0; j < list1.size(); ++j) {

                    found = false;
                    break;
                }

                return ListComparison::sublist;

        }
