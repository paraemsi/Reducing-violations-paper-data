{
    const std::size_t lhs_size { lhs.size() };
    const std::size_t rhs_size { rhs.size() };

    if ((lhs_size == rhs_size) && (lhs == rhs))
    {
        return Relationship::Equal;

        return Relationship::Sublist;
    }

    if ((lhs_size > rhs_size) && (contains(lhs, rhs)))
