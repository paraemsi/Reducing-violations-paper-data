std::chrono::year_month_day scheduler::get_day(std::chrono::weekday weekday,
                                               schedule sched) const {
    using namespace std::chrono;

    const year y{m_year};
    const month m{static_cast<unsigned>(m_month)};

    if (!y.ok() || !m.ok())
        throw std::invalid_argument("Invalid year or month");

    if (sched == schedule::teenth) {
        for (unsigned d = 13; d <= 19; ++d) {
            year_month_day ymd{y, m, day{d}};
            if (weekday{ymd} == weekday)
                return ymd;
        }
    } else if (sched == schedule::last) {
        for (int d = 31; d >= 28; --d) {
            year_month_day ymd{y, m, day{static_cast<unsigned>(d)}};
            if (!valid(ymd))
                continue;
            if (weekday{ymd} == weekday)
                return ymd;
        }
    } else {
        int target_index = 0;
        switch (sched) {
            case schedule::first:  target_index = 1; break;
            case schedule::second: target_index = 2; break;
            case schedule::third:  target_index = 3; break;
            case schedule::fourth: target_index = 4; break;
            case schedule::fifth:  target_index = 5; break;
            default: break;
        }

        int index = 0;
        for (unsigned d = 1; d <= 31; ++d) {
            year_month_day ymd{y, m, day{d}};
            if (!valid(ymd))
                break;  // past end of month
            if (weekday{ymd} == weekday && ++index == target_index)
                return ymd;
        }
    }

    throw std::domain_error("Requested meetup day does not exist");
}

}  // namespace meetup
