std::chrono::year_month_day scheduler::day(weekday wd, schedule sch) const {
    const std::chrono::year y{static_cast<int>(year_)};
    const std::chrono::month m{static_cast<unsigned>(month_)};

    const std::chrono::year_month_day first_ymd{y / m / std::chrono::day{1}};
    const std::chrono::sys_days first_sd{first_ymd};
    const std::chrono::weekday first_wd{first_sd};
    const std::chrono::weekday target_wd{to_chrono_weekday(wd)};

    if (sch == schedule::teenth) {
        const std::chrono::sys_days start_sd{y / m / std::chrono::day{13}};
        for (std::uint32_t d = 0U; d < 7U; ++d) {
            const std::chrono::sys_days cur{start_sd + std::chrono::days{static_cast<std::int64_t>(d)}};
            const std::chrono::weekday cur_wd{cur};
            if (cur_wd == target_wd) {
                return std::chrono::year_month_day{cur};
            }
        }
    } else if ((sch == schedule::first) || (sch == schedule::second) || (sch == schedule::third) || (sch == schedule::fourth)) {
        const std::uint32_t n = (sch == schedule::first) ? 0U
                              : ((sch == schedule::second) ? 1U
                              : ((sch == schedule::third) ? 2U : 3U));

        const std::uint32_t target_enc = static_cast<std::uint32_t>(target_wd.c_encoding());

        const std::uint32_t dayn = day1 + (7U * n);
        return std::chrono::year_month_day{y / m / std::chrono::day{dayn}};
    } else {
