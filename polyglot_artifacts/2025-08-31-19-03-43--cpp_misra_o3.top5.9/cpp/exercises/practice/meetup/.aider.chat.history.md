[[nodiscard]] constexpr bool is_leap_year(std::int32_t const year) noexcept
{
    /*  Parenthesise modulo expressions to satisfy MISRA precedence rule  */
    return ((static_cast<std::int32_t>((year % 400)) == 0) ||
            ((static_cast<std::int32_t>((year % 4)) == 0) &&
             (static_cast<std::int32_t>((year % 100)) != 0)));
}
