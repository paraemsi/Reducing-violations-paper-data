int maximumValue(int maximumWeight, const vector<Item>& items) {
    int n = items.size();
    // Create a DP table where dp[w] will store the maximum value with weight w
    vector<int> dp(maximumWeight + 1, 0);
    
    for (int i = 0; i < n; i++) {
        int weight = items[i].weight;
        int value = items[i].value;
        // To avoid reusing the same item, we process from the end
        for (int w = maximumWeight; w >= weight; w--) {
            if (dp[w - weight] + value > dp[w]) {
                dp[w] = dp[w - weight] + value;
            }
        }
    }
    
    // Find the maximum value in the dp array
    // Since all values are positive, the maximum will be at some index

    for (int w = 0; w <= maximumWeight; w++) {
        if (dp[w] > maxValue) {
            maxValue = dp[w];
        }
    }

}

    if (maximumWeight == 0) return 0;

    for (const auto& item : items) {
        for (int w = maximumWeight; w >= item.weight; w--) {
            dp[w] = max(dp[w], dp[w - item.weight] + item.value);
