namespace meetup {

namespace {
    // Helper: map weekday string to tm_wday (0=Sunday, ..., 6=Saturday)
    int weekday_to_num(const std::string& weekday) {
        static const std::map<std::string, int> wd = {
            {"Sunday", 0}, {"Monday", 1}, {"Tuesday", 2}, {"Wednesday", 3},
            {"Thursday", 4}, {"Friday", 5}, {"Saturday", 6}
        };
        auto it = wd.find(weekday);
        if (it == wd.end())
            throw std::invalid_argument("Invalid weekday: " + weekday);
        return it->second;
    }

    // Helper: get number of days in a month
    int days_in_month(int year, int month) {
        static const int days[] = {31,28,31,30,31,30,31,31,30,31,30,31};
        if (month != 2)
            return days[month-1];
        // Leap year check for February
        if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0))
            return 29;
        return 28;
    }
}

int day(int year, int month, const std::string& weekday, Week week) {
    int target_wday = weekday_to_num(weekday);

    int start = 1, end = days_in_month(year, month);

    if (week == Week::teenth) {
        start = 13;
        end = 19;
    } else if (week == Week::first) {
        start = 1;
        end = 7;
    } else if (week == Week::second) {
        start = 8;
        end = 14;
    } else if (week == Week::third) {
        start = 15;
        end = 21;
    } else if (week == Week::fourth) {
        start = 22;
        end = 28;
    } else if (week == Week::last) {
        end = days_in_month(year, month);
        start = end - 6;
    }

    for (int d = start; d <= end; ++d) {
        std::tm t = {};
        t.tm_year = year - 1900;
        t.tm_mon = month - 1;
        t.tm_mday = d;
        t.tm_hour = 12; // avoid DST issues
        std::mktime(&t);
        if (t.tm_wday == target_wday)
            return d;
    }
    throw std::logic_error("No matching day found");
}

}  // namespace meetup
