    // helper lambda to convert sys_days back to year_month_day
    auto to_ymd = [](const sys_days &sd) {
        return year_month_day{sd};
    };

    switch (s) {
    case schedule::teenth: {
        for (day d{13}; d <= day{19}; ++d) {
            year_month_day candidate = ym / d;
            if (weekday{candidate} == wd) {
                return candidate;
            }

        break; // should never fall through
    }
    case schedule::last: {
        year_month_day_last ymdl{ym / last};
        sys_days sd{ymdl};
        while (weekday{sd} != wd) {
            sd -= days{1};
        }
        return to_ymd(sd);
    }
    default: {
        // determine nth occurrence (0-indexed)
        int nth{};
        switch (s) {
        case schedule::first:  nth = 0; break;
        case schedule::second: nth = 1; break;
        case schedule::third:  nth = 2; break;
        case schedule::fourth: nth = 3; break;
        case schedule::fifth:  nth = 4; break;
        default:               nth = 0; break; // unreachable

        year_month_day first_of_month = ym / day{1};
        int diff = (wd - weekday{first_of_month} + days{7}).count() % 7;
        day target_day = day{1 + diff + 7 * nth};
        year_month_day candidate = ym / target_day;

        // If fifth occurrence spills to next month, behaviour is unspecified.
        return candidate;
    }
