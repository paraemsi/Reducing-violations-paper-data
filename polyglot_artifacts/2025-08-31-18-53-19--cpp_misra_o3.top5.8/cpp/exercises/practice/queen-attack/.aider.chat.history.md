bool chess_board::can_attack() const noexcept {
    const int col_diff{white_pos_.first - black_pos_.first};
    const int row_diff{white_pos_.second - black_pos_.second};

    if ((col_diff == 0) || (row_diff == 0)) {
        return true;
    }

    return (std::abs(col_diff) == std::abs(row_diff));
}
