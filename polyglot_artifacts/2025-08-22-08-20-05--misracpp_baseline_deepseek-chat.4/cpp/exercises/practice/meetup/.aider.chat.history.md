date scheduler::find_nth_weekday(int n, greg_weekday weekday) const {
    date first_of_month(year_, month_, 1);
    int days_to_add = (weekday - first_of_month.day_of_week() + 7) % 7;
    date result = first_of_month + days(days_to_add) + weeks(n - 1);
    if (result.month() != month_) {
        result -= weeks(1);
    }
    return result;
}

date scheduler::find_last_weekday(greg_weekday weekday) const {
    date last_of_month(year_, month_, gregorian_calendar::end_of_month_day(year_, month_));
    int days_to_subtract = (last_of_month.day_of_week() - weekday + 7) % 7;
    return last_of_month - days(days_to_subtract);
}
