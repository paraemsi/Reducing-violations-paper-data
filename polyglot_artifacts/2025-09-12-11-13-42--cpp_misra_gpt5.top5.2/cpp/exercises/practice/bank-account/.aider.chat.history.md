namespace Bankaccount {

Bankaccount::Bankaccount() noexcept
    : m_is_open(false), m_balance(static_cast<std::int64_t>(0)) {}

void Bankaccount::open() {
    std::lock_guard<std::mutex> lock(m_mutex);
    if (m_is_open) {
        throw std::logic_error("account already open");
    } else {
        m_is_open = true;
        m_balance = static_cast<std::int64_t>(0);
    }
}

void Bankaccount::close() {
    std::lock_guard<std::mutex> lock(m_mutex);
    if (!m_is_open) {
        throw std::logic_error("account already closed");
    } else {
        m_is_open = false;
    }
}

void Bankaccount::deposit(std::int64_t amount) {
    if ((amount <= static_cast<std::int64_t>(0))) {
        throw std::invalid_argument("deposit amount must be positive");
    }
    std::lock_guard<std::mutex> lock(m_mutex);
    if (!m_is_open) {
        throw std::logic_error("account is closed");
    } else {
        m_balance = m_balance + amount;
    }
}

void Bankaccount::withdraw(std::int64_t amount) {
    if ((amount <= static_cast<std::int64_t>(0))) {
        throw std::invalid_argument("withdraw amount must be positive");
    }
    std::lock_guard<std::mutex> lock(m_mutex);
    if (!m_is_open) {
        throw std::logic_error("account is closed");
    } else if ((m_balance < amount)) {
        throw std::logic_error("insufficient funds");
    } else {
        m_balance = m_balance - amount;
    }
}

std::int64_t Bankaccount::balance() const {
    std::lock_guard<std::mutex> lock(m_mutex);
    if (!m_is_open) {
        throw std::logic_error("account is closed");
    } else {
        return m_balance;
    }
}
