{
    return (((yr % 400) == 0) || (((yr % 4) == 0) && ((yr % 100) != 0)));
}

[[nodiscard]] constexpr std::uint32_t days_in_month(std::uint8_t mon,
                                                    std::int32_t yr) noexcept
{
    switch (mon)
    {
        case 1U:  case 3U:  case 5U:  case 7U:
        case 8U:  case 10U: case 12U:
        {
            return 31U;
        }
        case 4U:  case 6U:  case 9U:  case 11U:
        {
            return 30U;
        }
        case 2U:
        {
            return is_leap_year(yr) ? 29U : 28U;
        }
        default:
        {
            return 30U;   //  Defensive – should not occur for valid input
        }
    }

date meetup_day(std::int32_t         year_value,
                std::uint8_t         month_value,
                std::chrono::weekday weekday_req,
                schedule             selector)
{
    using namespace std::chrono;

    const sys_days first_of_month{
        year_month_day{year{year_value}, month{month_value}, day{1}}
    };

    //  Teenth: the 13th-19th always contains exactly one of each weekday
    if (selector == schedule::teenth)
    {
        sys_days current = first_of_month + days{12};          //  13th
        for (std::uint8_t i = 0U; i < 7U; ++i)
        {
            if (weekday{current} == weekday_req)
            {
                return year_month_day{current};
            }
            current += days{1};
        }
    }

    //  First-Fourth occurrences
    if ((selector == schedule::first)  || (selector == schedule::second) ||
        (selector == schedule::third)  || (selector == schedule::fourth))
    {
        const std::uint8_t target_index =
            static_cast<std::uint8_t>(selector) - 1U;  //  0-based
        sys_days       current = first_of_month;
        std::uint8_t   found   = 0U;

        while (true)
        {
            if (weekday{current} == weekday_req)
            {
                if (found == target_index)
                {
                    return year_month_day{current};
                }
                ++found;
            }
            current += days{1};
        }
    }

    //  Last occurrence
    if (selector == schedule::last)
    {
        const std::uint32_t dim = days_in_month(month_value, year_value);

        {
            current -= days{1};
        }
        return year_month_day{current};
    }

    return year_month_day{year{year_value}, month{month_value}, day{1}};
}
