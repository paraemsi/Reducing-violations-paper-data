std::vector<std::string> school::grade(int grade) const {
    std::vector<std::string> result;
    std::map<int, std::set<std::string>>::const_iterator it = roster_.find(grade);
    if (it != roster_.end()) {
        result.assign(it->second.begin(), it->second.end());
    }
    return result;
}
