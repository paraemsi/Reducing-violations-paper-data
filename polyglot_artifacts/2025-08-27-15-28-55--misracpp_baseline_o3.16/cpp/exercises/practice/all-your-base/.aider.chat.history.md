std::vector<int> convert(const std::vector<int>& digits,
                         int from_base,
                         int to_base) {
    // Validate bases
    if (from_base < 2 || to_base < 2) {
        throw std::invalid_argument("Bases must be >= 2.");
    }

    // Handle empty input as zero per exercise expectations
    if (digits.empty()) {
        return {0};
    }

    // Convert input digits to an integer value
    long long value = 0;
    for (int digit : digits) {
        if (digit < 0 || digit >= from_base) {
            throw std::invalid_argument("Digits out of range for input base.");
        }
        value = value * from_base + digit;
    }

    // Special-case zero
    if (value == 0) {
        return {0};
    }

    // Convert integer value to target base
    std::vector<int> result;
    while (value > 0) {
        result.push_back(static_cast<int>(value % to_base));
        value /= to_base;
    }
    std::reverse(result.begin(), result.end());
    return result;
}

}  // namespace all_your_base
