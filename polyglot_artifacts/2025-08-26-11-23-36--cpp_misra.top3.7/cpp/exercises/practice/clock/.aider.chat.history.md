    hour_t hour = m_hour;

    std::ostringstream oss;

        << std::setfill('0') << std::setw(2) << m_minute;
    return oss.str();

    minute_t total_minutes = (hour * MINUTES_PER_HOUR) + minute;

    hour = (total_minutes / MINUTES_PER_HOUR);
    minute = (total_minutes % MINUTES_PER_HOUR);
}
