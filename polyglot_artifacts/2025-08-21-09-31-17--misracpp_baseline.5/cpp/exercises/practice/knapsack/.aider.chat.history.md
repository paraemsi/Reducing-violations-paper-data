int knapsackMaxValue(const std::vector<Item>& items, int maxWeight)
{
    int n = items.size();
    std::vector<std::vector<int>> dp(n + 1, std::vector<int>(maxWeight + 1, 0));

    for (int i = 1; i <= n; ++i)
    {
        for (int w = 0; w <= maxWeight; ++w)
        {
            if (items[i - 1].weight > w)
            {
                dp[i][w] = dp[i - 1][w];
            }
            else
            {
                dp[i][w] = std::max(
                    dp[i - 1][w],
                    dp[i - 1][w - items[i - 1].weight] + items[i - 1].value
                );
            }
        }
    }
    return dp[n][maxWeight];
}

std::vector<int> knapsackSelectedItems(const std::vector<Item>& items, int maxWeight)
{
    int n = items.size();
    std::vector<std::vector<int>> dp(n + 1, std::vector<int>(maxWeight + 1, 0));

    // Fill DP table
    for (int i = 1; i <= n; ++i)
    {
        for (int w = 0; w <= maxWeight; ++w)
        {
            if (items[i - 1].weight > w)
            {
                dp[i][w] = dp[i - 1][w];
            }
            else
            {
                dp[i][w] = std::max(
                    dp[i - 1][w],
                    dp[i - 1][w - items[i - 1].weight] + items[i - 1].value
                );
            }
        }
    }

    // Backtrack to find selected items
    std::vector<int> selected;
    int w = maxWeight;
    for (int i = n; i >= 1; --i)
    {
        if (dp[i][w] != dp[i - 1][w])
        {
            selected.push_back(i - 1);
            w -= items[i - 1].weight;
        }
    }
    std::reverse(selected.begin(), selected.end());
    return selected;
}

} // namespace knapsack
