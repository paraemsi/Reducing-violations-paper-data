    // Return true if ym/d is a valid date
    bool valid_date(const std::chrono::year_month_day& ymd) {
        return ymd.ok();
    }

/**
 * Compute requested meetup day.
 */
std::chrono::year_month_day
scheduler::meetup_day(std::chrono::weekday wd, schedule sc) const
{
    using namespace std::chrono;

    const year y{m_year};
    const month m{m_month};

    if (!y.ok() || !m.ok()) {
        throw std::invalid_argument("Invalid year or month");
    }

    // Helper lambda that scans forward from start_day until weekday matches.
    auto find_from = [&](unsigned int start_day) -> year_month_day {
        for (unsigned int d = start_day; d <= 31; ++d) {
            year_month_day ymd{y, m, day{d}};
            if (!valid_date(ymd)) break;
            if (weekday{ymd} == wd) return ymd;
        }
        throw std::invalid_argument("Requested meetup day does not exist");
    };

    switch (sc) {
        case schedule::teenth:
            return find_from(13u);                       // 13 through 19 guarantee a hit
        case schedule::first:
            return find_from(1u);
        case schedule::second:
            return find_from(1u + 7u);                  // >= 8
        case schedule::third:
            return find_from(1u + 14u);                 // >= 15
        case schedule::fourth:
            return find_from(1u + 21u);                 // >= 22
        case schedule::last: {
            // Work backwards from end-of-month
            year_month_day_last last_day{year_month{y, m} / std::chrono::last};
            day d = last_day.day();
            for (unsigned int offset = 0; offset < 7; ++offset) {
                year_month_day candidate{y, m, d - days{offset}};
                if (weekday{candidate} == wd) return candidate;
            }
            break;
        }
    }
    throw std::invalid_argument("Requested meetup day does not exist");
}

}  // namespace meetup
