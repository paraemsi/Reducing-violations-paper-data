#### A circular buffer, cyclic buffer or ring buffer is a data structure that uses a single, fixed-size buffer as if it were connected end-to-end.  
####   
#### A circular buffer first starts empty and of some predefined length.  
#### For example, this is a 7-element buffer:  
####   
#### ```text  
#### [ ][ ][ ][ ][ ][ ][ ]  
#### ```  
####   
#### Assume that a 1 is written into the middle of the buffer (exact starting location does not matter in a circular buffer):  
####   
#### ```text  
#### [ ][ ][ ][1][ ][ ][ ]  
#### ```  
####   
#### Then assume that two more elements are added — 2 & 3 — which get appended after the 1:  
####   
#### ```text  
#### [ ][ ][ ][1][2][3][ ]  
#### ```  
####   
#### If two elements are then removed from the buffer, the oldest values inside the buffer are removed.  
#### The two elements removed, in this case, are 1 & 2, leaving the buffer with just a 3:  
####   
#### ```text  
#### [ ][ ][ ][ ][ ][3][ ]  
#### ```  
####   
#### If the buffer has 7 elements then it is completely full:  
####   
#### ```text  
#### [5][6][7][8][9][3][4]  
#### ```  
####   
#### When the buffer is full an error will be raised, alerting the client that further writes are blocked until a slot becomes free.  
####   
#### When the buffer is full, the client can opt to overwrite the oldest data with a forced write.  
#### In this case, two more elements — A & B — are added and they overwrite the 3 & 4:  
####   
#### ```text  
#### [5][6][7][8][9][A][B]  
#### ```  
####   
#### 3 & 4 have been replaced by A & B making 5 now the oldest data in the buffer.  
#### Finally, if two elements are removed then what would be returned is 5 & 6 yielding the buffer:

#### [ ][ ][7][8][9][A][B]

#### ```

#include <cstddef>

#include <vector>

/*
 * Template implementation of a circular (ring) buffer.
 *
 * The buffer raises std::runtime_error if a write is attempted while full.
 * The overwrite() member inserts the new element, discarding the oldest one
 * whenever space is unavailable.
 */
template<typename T>
class CircularBuffer
{
public:
    explicit CircularBuffer(std::size_t capacity) :
        m_storage(capacity),
        m_head(static_cast<std::size_t>(0U)),
        m_tail(static_cast<std::size_t>(0U)),
        m_full(false)
    {
        if (capacity == static_cast<std::size_t>(0U)) {
            throw std::invalid_argument("Capacity must be greater than zero");
        }
    }

    void write(const T& item)
    {
        if (full()) {
            throw std::runtime_error("Buffer is full");
        }

        m_storage[m_head] = item;
        m_head = advance(m_head);

        if (m_head == m_tail) {
            m_full = true;
        }
    }

    void overwrite(const T& item)
    {
        if (full()) {
            /* When full, insertion discards the oldest element first. */
            m_storage[m_head] = item;
            m_head = advance(m_head);
            m_tail = advance(m_tail);
        } else {
            write(item);
        }
    }

    [[nodiscard]] T read()
    {
        if (empty()) {
            throw std::runtime_error("Buffer is empty");
        }

        const T value = m_storage[m_tail];
        m_full = false;
        m_tail = advance(m_tail);
        return value;

    {
        m_head = m_tail;
        m_full = false;
