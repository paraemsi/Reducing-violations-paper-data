     *
     *  dp[i][w] holds the maximum value achievable using the first i items
     *  without exceeding total weight w.
     */

    std::vector<std::vector<std::uint32_t>> dp(

        std::vector<std::uint32_t>(capacity + static_cast<std::uint32_t>(1U), 0U));

    for (std::size_t i { 1U }; i <= item_count; ++i)
    {
        const Item& current_item { items[i - static_cast<std::size_t>(1U)] };

        for (std::uint32_t w { 0U }; w <= capacity; ++w)
        {
            if (w >= current_item.weight)
            {
                const std::uint32_t without_item { dp[i - static_cast<std::size_t>(1U)][w] };
                const std::uint32_t with_item {
                    static_cast<std::uint32_t>(
                        dp[i - static_cast<std::size_t>(1U)]

                dp[i][w] = std::max(without_item, with_item);
            }
            else
            {
                dp[i][w] = dp[i - static_cast<std::size_t>(1U)][w];
            }
        }
    }
