    }

}

std::uint8_t weekday_to_num(const std::string& weekday)
{
    // 0 = Sunday, 1 = Monday, ..., 6 = Saturday
    static const std::map<std::string, std::uint8_t> lookup = {
        {"Sunday", 0U}, {"Monday", 1U}, {"Tuesday", 2U}, {"Wednesday", 3U},
        {"Thursday", 4U}, {"Friday", 5U}, {"Saturday", 6U}
    };
    auto it = lookup.find(weekday);
    if (it == lookup.end()) {
        throw std::invalid_argument("Invalid weekday: " + weekday);
    }
    return it->second;
}

day_t get_weekday_of_date(year_t year, month_t month, day_t day)
{
    // Zeller's congruence, 0 = Saturday, 1 = Sunday, ..., 6 = Friday
    year_t y = year;
    month_t m = month;
    if (m < 3U) {
        m = static_cast<month_t>(m + 12U);
        y = static_cast<year_t>(y - 1);

    if (h == 0U) {

{
    // 0 = Sunday, 1 = Monday, ..., 6 = Saturday
