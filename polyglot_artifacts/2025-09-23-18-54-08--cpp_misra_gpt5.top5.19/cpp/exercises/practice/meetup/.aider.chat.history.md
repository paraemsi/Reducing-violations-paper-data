scheduler::scheduler(std::int32_t year, std::uint32_t month) noexcept
    : _year{std::chrono::year{static_cast<int>(year)}},
      _month{std::chrono::month{static_cast<unsigned int>(month)}} {}

std::chrono::year_month_day scheduler::find_teenth(std::chrono::weekday wd) const
{
    for (std::uint32_t d = 13U; (d <= 19U); ++d) {
        const std::chrono::year_month_day ymd{
            _year,
            _month,
            std::chrono::day{static_cast<unsigned int>(d)}
        };
        const std::chrono::weekday w{std::chrono::sys_days{ymd}};
        if ((w == wd)) {
            return ymd;
        }
    }
    // Fallback; logically unreachable because there is always exactly one teenth for each weekday
    return std::chrono::year_month_day{
        _year,
        _month,
        std::chrono::day{static_cast<unsigned int>(19U)}
    };
}

std::chrono::year_month_day scheduler::find_nth(std::chrono::weekday wd, std::uint32_t n) const
{
    const std::chrono::year_month_weekday ymwd{
        _year,
        _month,
        std::chrono::weekday_indexed{wd, static_cast<unsigned int>(n)}
    };
    return std::chrono::year_month_day{std::chrono::sys_days{ymwd}};
}

std::chrono::year_month_day scheduler::find_last(std::chrono::weekday wd) const
{
    const std::chrono::year_month_weekday_last ymwdl{
        _year,
        _month,
        std::chrono::weekday_last{wd}
    };
    return std::chrono::year_month_day{std::chrono::sys_days{ymwdl}};
