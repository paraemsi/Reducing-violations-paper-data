scheduler::day(schedule week, std::chrono::weekday weekday) const

    /* Determine the starting day for the search */
    std::uint32_t start_day_value{1U};

    switch (week) {
        case schedule::first:  { start_day_value = 1U;  break; }
        case schedule::second: { start_day_value = 8U;  break; }
        case schedule::third:  { start_day_value = 15U; break; }
        case schedule::fourth: { start_day_value = 22U; break; }
        case schedule::teenth: { start_day_value = 13U; break; }
        case schedule::last:   { /* handled separately below */ break; }
        default:               { break; }   /* unreachable */
    }

    /* Use sys_days to perform day arithmetic — year_month_day lacks -=/+= days */
    if (week == schedule::last) {
        /* First day of the current month */
        const std::chrono::year_month_day first_of_month{
            m_year,
            m_month,
            std::chrono::day{1U}
        };

        /* First day of the next month */
        const std::chrono::year_month_day first_of_next_month{
            (first_of_month.year() + std::chrono::months{1}) /
            (first_of_month.month() + std::chrono::months{1}) /
            std::chrono::day{1U}
        };

        /* Convert to sys_days, step one day back to reach last day of current month */
        std::chrono::sys_days candidate{
            std::chrono::sys_days{first_of_next_month} - std::chrono::days{1}
