void school::add(const std::string& name, grade_t grade)
{
    /*  First, ensure the student is not already enrolled in another grade.
     *  If found, remove the old entry so the student can be re-assigned.   */
    for(auto& pair : m_data)
    {
        const auto iter = pair.second.find(name);
        if((iter != pair.second.end()))
        {
            if(pair.first == grade)
            {
                /* Student is already in the correct grade → nothing to do. */
                return;
            }
            pair.second.erase(iter);
            if(pair.second.empty())
            {
                /*  Erase empty grade entry to keep the map tidy.
                 *  The iterator returned by erase would be end(), so break. */
                m_data.erase(pair.first);
            }
            break;
        }
    }

    /* Insert the student into the requested grade. */
    (void)m_data[grade].insert(name);
}

std::vector<std::string> school::grade(grade_t grade) const
{
    const auto iter = m_data.find(grade);
    if(iter == m_data.end())
    {
        return {};
    }
    return std::vector<std::string>(iter->second.begin(), iter->second.end());
}

std::map<grade_t, std::vector<std::string>> school::roster() const
{
    std::map<grade_t, std::vector<std::string>> result{};
    for(const auto& pair : m_data)
    {
        result.emplace(pair.first,
                       std::vector<std::string>(pair.second.begin(),
                                                pair.second.end()));
    }
    return result;
}

}  // namespace grade_school
