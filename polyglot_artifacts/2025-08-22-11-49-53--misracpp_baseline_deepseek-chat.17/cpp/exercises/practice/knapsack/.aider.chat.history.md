int maximumValue(int maximumWeight, const vector<Item>& items) {
    if (items.empty() || maximumWeight == 0) {
        return 0;
    }
    
    int n = items.size();
    // Using a 1D DP array to save space
    vector<int> dp(maximumWeight + 1, 0);
    
    for (int i = 0; i < n; i++) {
        int weight = items[i].weight;
        int value = items[i].value;
        // Process from the end to prevent reusing the same item
        for (int w = maximumWeight; w >= weight; w--) {
            dp[w] = max(dp[w], dp[w - weight] + value);
        }
    }
    
    return dp[maximumWeight];
}

} // namespace knapsack
