};

template <typename T>
inline bool contains_subsequence(const std::vector<T>& haystack, const std::vector<T>& needle)
{
    using size_type = typename std::vector<T>::size_type;

    if ((needle.size() == static_cast<size_type>(0U))) {
        return true;
    }

    if ((haystack.size() < needle.size())) {
        return false;
    }

    const size_type last_start = (haystack.size() - needle.size());

    for (size_type i = static_cast<size_type>(0U); i <= last_start; ++i) {
        bool match = true;

        for (size_type j = static_cast<size_type>(0U); j < needle.size(); ++j) {
            if (!((haystack[(i + j)]) == needle[j])) {
                match = false;
                break;
            }
        }

        if (match) {
            return true;
        }
    }

    return false;
}

template <typename T>
inline relation compare(const std::vector<T>& a, const std::vector<T>& b)
{
    if ((a.size() == b.size())) {
        if (std::equal(a.begin(), a.end(), b.begin(), b.end())) {
            return relation::equal;
        } else {
            return relation::unequal;
        }
    } else if ((a.size() > b.size())) {

            return relation::superlist;
        } else {
            return relation::unequal;
        }
    } else {
        if (contains_subsequence(b, a)) {

        } else {
            return relation::unequal;
        }
    }
}
