#### Given the position of two queens on a chess board, indicate whether or not they are positioned so that they can attack each other.

bool chess::is_valid_position(position_t pos) noexcept
{
    return ((pos.first  >= static_cast<coordinate_t>(0)) &&
            (pos.first  <  static_cast<coordinate_t>(8)) &&
            (pos.second >= static_cast<coordinate_t>(0)) &&
            (pos.second <  static_cast<coordinate_t>(8)));
}

chess::chess(position_t white, position_t black)
    : white_position_{white},
      black_position_{black}
{
    if ((!is_valid_position(white)) || (!is_valid_position(black)) || (white == black))
    {
        throw std::domain_error("Invalid queen position");
    }
}

position_t chess::white() const noexcept
{
    return white_position_;
}

position_t chess::black() const noexcept
{
    return black_position_;
}

bool chess::can_attack() const noexcept
{
    const std::int16_t row_diff =
        static_cast<std::int16_t>(white_position_.first) -
        static_cast<std::int16_t>(black_position_.first);

    const std::int16_t col_diff =
        static_cast<std::int16_t>(white_position_.second) -
        static_cast<std::int16_t>(black_position_.second);

    return ((row_diff == static_cast<std::int16_t>(0)) ||
            (col_diff == static_cast<std::int16_t>(0)) ||
            (row_diff == col_diff) ||
            (row_diff == static_cast<std::int16_t>(-col_diff)));
}

}  // namespace queen_attack
