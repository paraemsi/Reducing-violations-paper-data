KindergartenGarden::KindergartenGarden(const std::string& diagram) {
    std::size_t newline_pos = diagram.find('\n');
    if ((newline_pos != std::string::npos)) {
        row1 = diagram.substr(0U, newline_pos);
        row2 = diagram.substr((newline_pos + 1U));
    } else {
        // Assume valid input with newline; if not, row1 gets the whole string and row2 empty
        row1 = diagram;
        row2 = "";
    }
}

std::vector<std::string> KindergartenGarden::plants(const std::string& child) const {
    std::vector<std::string>::const_iterator it = std::find(children.begin(), children.end(), child);
    if ((it == children.end())) {
        return {};
    }
    std::size_t index = static_cast<std::size_t>(it - children.begin());
    std::size_t pos1 = (2U * index);
    std::size_t pos2 = (pos1 + 1U);
    std::vector<std::string> result;
    if (((pos2 < row1.size()) && (pos2 < row2.size()))) {
        result.push_back(plant_name(row1[pos1]));
        result.push_back(plant_name(row1[pos2]));
        result.push_back(plant_name(row2[pos1]));
        result.push_back(plant_name(row2[pos2]));
    }
    return result;
}

std::string KindergartenGarden::plant_name(char plant_code) const {
    if ((plant_code == 'G')) {
        return "grass";
    } else if ((plant_code == 'C')) {
        return "clover";
    } else if ((plant_code == 'R')) {
        return "radishes";
    } else if ((plant_code == 'V')) {
        return "violets";
    } else {
        return "";
    }
}

}  // namespace kindergarten_garden
