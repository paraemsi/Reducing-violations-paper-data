    year_month_day day(const year_month& ym,
                       const weekday& wd,
                       schedule s) {
        switch (s) {
            case schedule::first:
            case schedule::second:
            case schedule::third:
            case schedule::fourth: {
                unsigned index = 0;
                switch (s) {
                    case schedule::first:  index = 1; break;
                    case schedule::second: index = 2; break;
                    case schedule::third:  index = 3; break;
                    case schedule::fourth: index = 4; break;
                    default:               index = 0; break;
                }
                weekday_indexed wdi{wd, index};
                auto ymwd = ym.year() / ym.month() / wdi;      // year_month_weekday
                return year_month_day{sys_days{ymwd}};
            }
            case schedule::last: {
                weekday_last wdl{wd};
                auto ymwl = ym.year() / ym.month() / wdl;      // year_month_weekday_last
                return year_month_day{sys_days{ymwl}};
            }
            case schedule::teenth: {
                for (unsigned d = 13; d <= 19; ++d) {
                    year_month_day ymd{ym.year(), ym.month(), day{d}};
                    if (weekday{sys_days{ymd}} == wd) {
                        return ymd;
                    }
                }
                break; // Should never be reached
            }
        }
        throw std::logic_error("Invalid schedule");
    }

}  // namespace meetup
