int weekday_str_to_num(const std::string& weekday) {
    // 0 = Sunday, 1 = Monday, ..., 6 = Saturday
    static const std::array<std::string, 7> names = {
        "Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"
    };
    auto it = std::find_if(names.begin(), names.end(), [&](const std::string& s){
        return std::equal(s.begin(), s.end(), weekday.begin(), weekday.end(),
                          [](char a, char b){ return std::tolower(a)==std::tolower(b); });
    });
    if (it == names.end())
        throw std::invalid_argument("Invalid weekday: " + weekday);
    return static_cast<int>(std::distance(names.begin(), it));
}

unsigned days_in_month(int year, int month) {
    using namespace std::chrono;
    auto y = std::chrono::year{year};
    auto m = std::chrono::month{static_cast<unsigned>(month)};

    auto last = sys_days{next_month} - std::chrono::days{1};

}
