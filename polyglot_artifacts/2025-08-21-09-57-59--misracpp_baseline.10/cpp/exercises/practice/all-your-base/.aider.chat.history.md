std::vector<int> convert(int from_base, const std::vector<int>& digits, int to_base) {
    // Validate bases
    if (from_base < 2) {
        throw std::invalid_argument("from_base must be >= 2");
    }
    if (to_base < 2) {
        throw std::invalid_argument("to_base must be >= 2");
    }
    // Validate digits
    if (!digits.empty()) {
        bool leading_zero = digits.size() > 1 && digits[0] == 0;
        if (leading_zero) {
            throw std::invalid_argument("digits may not contain leading zeros");
        }
        for (int d : digits) {
            if (d < 0 || d >= from_base) {
                throw std::invalid_argument("all digits must satisfy 0 <= d < from_base");
            }
        }
    }

    // Convert input digits to integer value
    int value = 0;
    for (int d : digits) {
        value = value * from_base + d;
    }

    // Special case: zero
    if (digits.empty() || value == 0) {
        return {0};
    }

    // Convert integer value to output base
    std::vector<int> result;
    while (value > 0) {
        result.push_back(value % to_base);
        value /= to_base;
    }
    std::reverse(result.begin(), result.end());
    return result;
}

}  // namespace all_your_base
