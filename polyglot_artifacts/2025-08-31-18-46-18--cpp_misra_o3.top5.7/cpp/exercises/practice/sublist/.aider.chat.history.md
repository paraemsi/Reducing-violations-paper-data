{
    Equal,      // Lists are identical
    Sublist,    // First list is contained within the second
    Superlist,  // First list contains the second
    Unequal     // None of the above
};

/*
 * Determine the relation between listA and listB.
 * Requirements:
 *  - T must be equality-comparable.
 */
template <typename T>
inline Relation compare(const std::vector<T>& listA, const std::vector<T>& listB)
{
    /* Equality check */
    if (listA == listB)
    {
        return Relation::Equal;
    }

    /* listA shorter ⇒ potential sublist */
    if (listA.size() < listB.size())
    {
        if (std::search(listB.begin(), listB.end(), listA.begin(), listA.end()) != listB.end())
        {
            return Relation::Sublist;
        }
    }
    /* listA longer ⇒ potential superlist */
    else if (listA.size() > listB.size())
    {
        if (std::search(listA.begin(), listA.end(), listB.begin(), listB.end()) != listA.end())
        {
            return Relation::Superlist;
        }
    }

    return Relation::Unequal;
}

}  // namespace sublist
