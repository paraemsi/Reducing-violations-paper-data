template <typename T>
inline bool is_sublist(const std::vector<T>& small, const std::vector<T>& big) {
    if (small.empty()) {
        return true;
    }
    if (small.size() > big.size()) {
        return false;
    }
    auto it = std::search(big.begin(), big.end(), small.begin(), small.end());
    return it != big.end();
}

template <typename T>
inline Comparison check_lists(const std::vector<T>& a, const std::vector<T>& b) {

    }

        return Comparison::sublist;
    }
    if (is_sublist(b, a)) {
        return Comparison::superlist;
    }
    return Comparison::unequal;
}
