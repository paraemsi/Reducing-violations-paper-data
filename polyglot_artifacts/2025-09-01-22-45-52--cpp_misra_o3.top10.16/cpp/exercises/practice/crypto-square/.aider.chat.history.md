#### Given an English text, output the encoded version of that text.

#### First, the input is normalized: the spaces and punctuation are removed from the English text and the message is down-cased.

####   
#### If `c` is the number of columns and `r` is the number of rows, then for the rectangle `r` x `c` find the smallest possible integer `c` such that:  
####   
#### - `r * c >= length of message`,

/*
 * Normalise the input string by
 *   – removing all non-alphanumeric characters, and
 *   – converting the remaining characters to lower-case.
 */
auto normalize(const std::string& input) -> std::string
{
    const std::locale loc {};
    std::string output {};
    output.reserve(input.size());

    for(char ch : input)
    {
        if(std::isalnum(static_cast<unsigned char>(ch), loc) != 0)
        {
            output.push_back(
                static_cast<char>(
                    std::tolower(static_cast<unsigned char>(ch), loc)));
        }
    }
    return output;
}

} // unnamed namespace

auto cipher(const std::string& input) -> std::string
{
    const std::string normalized { normalize(input) };
    const std::uint64_t len { static_cast<std::uint64_t>(normalized.size()) };

    if(len == static_cast<std::uint64_t>(0U))
    {
        return std::string {};
    }

    const double root { std::sqrt(static_cast<double>(len)) };
    std::uint64_t rows { static_cast<std::uint64_t>(root) };
    std::uint64_t columns { static_cast<std::uint64_t>(std::ceil(root)) };

    const std::uint64_t area { rows * columns };
    if(area < len)
    {
        ++rows;
    }

    std::vector<std::string> chunks {};
    chunks.reserve(columns);

    for(std::uint64_t col = static_cast<std::uint64_t>(0U); col < columns; ++col)
    {
        std::string chunk {};
        chunk.reserve(rows);

        for(std::uint64_t row = static_cast<std::uint64_t>(0U); row < rows; ++row)
        {
            const std::uint64_t index { (row * columns) + col };
            if(index < len)
            {
                chunk.push_back(
                    normalized[static_cast<std::size_t>(index)]);
            }
            else
            {
                chunk.push_back(' ');
            }
        }
        chunks.push_back(chunk);
    }

    std::ostringstream oss {};
    for(std::uint64_t i = static_cast<std::uint64_t>(0U); i < columns; ++i)
    {
        oss << chunks[static_cast<std::size_t>(i)];
        if(((i + static_cast<std::uint64_t>(1U)) < columns))
        {
            oss << ' ';
        }
    }

    return oss.str();
}
