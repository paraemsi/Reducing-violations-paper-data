static void validate_input(const std::vector<int>& digits,
                           int from_base,
                           int to_base)
{
    if (from_base < 2 || to_base < 2) {
        throw std::invalid_argument("bases must be >= 2");
    }
    if (digits.empty()) {
        throw std::invalid_argument("digits may not be empty");
    }
    bool all_zero = true;
    for (int d : digits) {
        if (d < 0 || d >= from_base) {
            throw std::invalid_argument("digit out of range for input base");
        }
        if (d != 0) {
            all_zero = false;
        }
    }
    // allow an all-zero input like {0}
    if (all_zero && digits.size() > 1) {
        // Leading zeros beyond the single zero are invalid.
        throw std::invalid_argument("invalid leading zeros");
    }
}

static unsigned long long to_base10(const std::vector<int>& digits,
                                    int from_base)
{
    unsigned long long value = 0;
    for (int d : digits) {
        value = value * static_cast<unsigned long long>(from_base)
              + static_cast<unsigned long long>(d);
    }
    return value;
}

static std::vector<int> from_base10(unsigned long long value, int to_base)
{
    if (value == 0) {
        return {0};
    }

    std::vector<int> out;
    while (value > 0) {
        out.push_back(static_cast<int>(value % to_base));
        value /= to_base;
    }
    std::reverse(out.begin(), out.end());
    return out;
}

std::vector<int> convert(const std::vector<int>& digits,
                         int from_base,
                         int to_base)
{
    validate_input(digits, from_base, to_base);
    const auto value = to_base10(digits, from_base);
    return from_base10(value, to_base);
}

}  // namespace all_your_base
