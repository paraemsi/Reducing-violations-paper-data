// Return the requested meetup day as a std::chrono::year_month_day
year_month_day scheduler::day(weekday wd, schedule s) const
{
    std::vector<year_month_day> matches;

    // First day of the month
    year_month_day first_day{year_, month_, day{1}};
    sys_days cur{first_day};

    // One past the last day of the month (i.e. first of next month)
    sys_days end{(first_day + months{1})};

    // Gather every date in the month that matches the target weekday
    for (; cur < end; cur += days{1}) {
        year_month_day ymd{cur};
        if (weekday{ymd} == wd) {
            matches.push_back(ymd);
        }
    }

    // Handle the various scheduling keywords
    switch (s) {
        case schedule::first:   return matches.at(0);
        case schedule::second:  return matches.at(1);
        case schedule::third:   return matches.at(2);
        case schedule::fourth:  return matches.at(3);
        case schedule::fifth:   return matches.at(4);   // may throw if no 5th
        case schedule::last:    return matches.back();
        case schedule::teenth:
            for (const auto& ymd : matches) {
                unsigned d = static_cast<unsigned>(ymd.day());
                if (d >= 13 && d <= 19) {
                    return ymd;
                }
            }
            break;  // fall-through to invalid return below
    }

    // If the request cannot be satisfied return an invalid date
    return year_month_day{};
}

}  // namespace meetup
