std::vector<int> convert(const std::vector<int>& digits, int from_base, int to_base) {
    // Validate bases.
    if (from_base < 2) {
        throw std::invalid_argument("input base must be >= 2");
    }
    if (to_base < 2) {
        throw std::invalid_argument("output base must be >= 2");
    }

    // Validate digits.
    if (digits.empty()) {
        throw std::invalid_argument("no digits provided");
    }
    for (int d : digits) {
        if (d < 0 || d >= from_base) {
            throw std::invalid_argument("digit out of range for input base");
        }
    }

    // Remove leading zeros; if all digits are zero return {0}.
    auto first_non_zero = std::find_if(digits.begin(), digits.end(),
                                       [](int d) { return d != 0; });
    if (first_non_zero == digits.end()) {
        return {0};
    }

    // Working copy of the number in the source base.
    std::vector<int> value(first_non_zero, digits.end());

    std::vector<int> result;               // Digits of the number in the target base (LSB first).
    while (!value.empty()) {
        std::vector<int> quotient;         // Next value in source base after division.
        int remainder = 0;

        for (int d : value) {
            long long accumulator = static_cast<long long>(remainder) * from_base + d;
            int q_digit  = static_cast<int>(accumulator / to_base);
            remainder    = static_cast<int>(accumulator % to_base);
            if (!quotient.empty() || q_digit != 0) {
                quotient.push_back(q_digit);
            }
        }
        result.push_back(remainder);
        value.swap(quotient);
    }

    std::reverse(result.begin(), result.end());
    return result;
}

}  // namespace all_your_base
