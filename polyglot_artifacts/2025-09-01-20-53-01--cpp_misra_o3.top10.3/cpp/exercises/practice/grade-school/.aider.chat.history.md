auto school::add(const std::string& name, grade_t grade) -> void
{
    for (auto const& grade_students_pair : m_roster) {
        auto const& students = grade_students_pair.second;
        if (std::find(students.begin(), students.end(), name) != students.end()) {
            throw std::domain_error{"Student already in roster"};
        }
    }

    auto& students = m_roster[grade];
    students.push_back(name);
    std::sort(students.begin(), students.end());
}

auto school::grade(grade_t grade) const -> std::vector<std::string>
{
    auto const it = m_roster.find(grade);
    if ((it != m_roster.end())) {
        return it->second;
    } else {
        return {};
    }
}
