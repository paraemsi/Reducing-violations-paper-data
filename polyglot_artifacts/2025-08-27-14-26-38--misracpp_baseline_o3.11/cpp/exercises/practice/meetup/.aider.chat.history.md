// helper: nth weekday (1–4)
std::chrono::year_month_day scheduler::nth_weekday(int n, std::chrono::weekday wd) const {
    if (n < 1 || n > 4) {
        throw std::invalid_argument("nth_weekday: n must be 1..4");
    }
    using namespace std::chrono;

    const sys_days first_day = sys_days{year_month_day{year{year_}, month{month_}, day{1}}};
    const weekday first_wd{first_day};

    const days diff = (wd - first_wd + days{7}) % days{7};
    const sys_days result = first_day + diff + days{7 * (n - 1)};

    return year_month_day{result};
}

// helper: last weekday
std::chrono::year_month_day scheduler::last_weekday(std::chrono::weekday wd) const {
    using namespace std::chrono;

    const sys_days last_day =
        sys_days{year_month_day_last{year{year_}, month_day_last{month{month_}}}};
    const weekday last_wd{last_day};

    const days diff = (last_wd - wd + days{7}) % days{7};
    const sys_days result = last_day - diff;

    return year_month_day{result};
}
