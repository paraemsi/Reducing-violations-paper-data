scheduler::scheduler(int year, unsigned int month)
    : _year{std::chrono::year{year}}, _month{std::chrono::month{month}} {}

/* Compute the exact day of the month that matches the requested weekday and
   schedule (first, second, …, teenth, last). */
std::chrono::year_month_day scheduler::get_day(const std::chrono::weekday& wd,
                                               schedule s) const {
    using namespace std::chrono;

    year_month ym{_year, _month};
    std::vector<year_month_day> matches;

    /* Collect every day in the month that lands on the requested weekday. */
    for (unsigned int d = 1;; ++d) {
        year_month_day ymd = ym / day{d};
        if (!ymd.ok())          // We ran past the valid range for this month.
            break;
        if (weekday{sys_days{ymd}} == wd)
            matches.push_back(ymd);
    }

    switch (s) {
        case schedule::first:   return matches.at(0);
        case schedule::second:  return matches.at(1);
        case schedule::third:   return matches.at(2);
        case schedule::fourth:  return matches.at(3);
        case schedule::last:    return matches.back();
        case schedule::teenth:
            for (const auto& ymd : matches) {
                unsigned int d = static_cast<unsigned int>(unsigned(ymd.day()));
                if (d >= 13 && d <= 19)
                    return ymd;
            }
            break;
    }

    /* If we get here, the combination was impossible for the given month. */
    throw std::logic_error("Requested meetup day does not exist in this month");
}

}  // namespace meetup

    : _year{std::chrono::year{year}}, _month{std::chrono::month{month}} {}

/* Compute the exact day of the month that matches the requested weekday and
   schedule (first, second, …, teenth, last). */
std::chrono::year_month_day scheduler::get_day(const std::chrono::weekday& wd,
                                               schedule s) const {
    using namespace std::chrono;

    year_month ym{_year, _month};
    std::vector<year_month_day> matches;

    /* Collect every day in the month that lands on the requested weekday. */
    for (unsigned int d = 1;; ++d) {
        year_month_day ymd = ym / day{d};
        if (!ymd.ok())          // We ran past the valid range for this month.
            break;
        if (weekday{sys_days{ymd}} == wd)
            matches.push_back(ymd);
    }

    switch (s) {
        case schedule::first:   return matches.at(0);
        case schedule::second:  return matches.at(1);
        case schedule::third:   return matches.at(2);
        case schedule::fourth:  return matches.at(3);
        case schedule::last:    return matches.back();
        case schedule::teenth:
            for (const auto& ymd : matches) {
                unsigned int d = unsigned(ymd.day());
                if (d >= 13 && d <= 19)
                    return ymd;
            }
            break;
    }

    /* If we get here, the combination was impossible for the given month. */
