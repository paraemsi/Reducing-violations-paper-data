namespace grade_school {

bool school::add(const std::string& name, int grade) {
    // Reject duplicate student names
    if (m_student_grade.count(name) != 0) {
        return false;
    }
    m_student_grade[name] = grade;
    m_roster[grade].insert(name);
    return true;
}

std::vector<std::string> school::grade(int grade) const {
    auto it = m_roster.find(grade);
    if (it == m_roster.end()) {
        return {};
    }
    return std::vector<std::string>(it->second.begin(), it->second.end());
}

std::map<int, std::vector<std::string>> school::roster() const {
    std::map<int, std::vector<std::string>> result;
    for (const auto& [g, students] : m_roster) {
        result.emplace(g, std::vector<std::string>(students.begin(), students.end()));
    }
    return result;
}

}  // namespace grade_school
