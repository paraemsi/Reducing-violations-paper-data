    return black_;

bool chess_board::can_attack() const noexcept {
    const coordinate_t wx = white_.first;
    const coordinate_t wy = white_.second;
    const coordinate_t bx = black_.first;
    const coordinate_t by = black_.second;

    const bool same_row = (wy == by);
    const bool same_col = (wx == bx);

    using wide_t = std::uint16_t;
    const wide_t wx_w = static_cast<wide_t>(wx);
    const wide_t wy_w = static_cast<wide_t>(wy);
    const wide_t bx_w = static_cast<wide_t>(bx);
    const wide_t by_w = static_cast<wide_t>(by);

    const wide_t dx = ((wx_w > bx_w) ? (wx_w - bx_w) : (bx_w - wx_w));
    const wide_t dy = ((wy_w > by_w) ? (wy_w - by_w) : (by_w - wy_w));

    const bool same_diag = (dx == dy);

    return ((same_row || same_col) || same_diag);
}
