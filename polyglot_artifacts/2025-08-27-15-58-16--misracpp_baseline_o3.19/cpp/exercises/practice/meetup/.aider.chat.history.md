namespace meetup {

using namespace std::chrono;

scheduler::scheduler(int year, int month)
    : m_year{year}, m_month{static_cast<unsigned>(month)} {}

// helper: days in current month
static unsigned days_in_month(year y, month m)
{
    // first day of following month, then step back one day
    year    next_year  = (m == month{12}) ? y + years{1} : y;
    month   next_month = (m == month{12}) ? month{1}      : m + months{1};

    sys_days first_next_month = sys_days{year_month_day{next_year / next_month / 1}};
    sys_days last_this_month  = first_next_month - days{1};

    return static_cast<unsigned>(year_month_day{last_this_month}.day());
}

year_month_day scheduler::get_day(weekday target_weekday, schedule sched) const
{
    // first day of the month
    sys_days first = sys_days{year_month_day{m_year / m_month / 1}};
    weekday  first_wd{first};

    unsigned day_num{0};

    switch (sched) {
        case schedule::teenth: {
            for (unsigned d = 13; d <= 19; ++d) {
                sys_days dt = sys_days{year_month_day{m_year / m_month / day{d}}};
                if (weekday{dt} == target_weekday) {
                    day_num = d;
                    break;
                }
            }
            break;
        }

        case schedule::first:
        case schedule::second:
        case schedule::third:
        case schedule::fourth: {
            int ordinal = 0;
            if (sched == schedule::first)   ordinal = 1;
            if (sched == schedule::second)  ordinal = 2;
            if (sched == schedule::third)   ordinal = 3;
            if (sched == schedule::fourth)  ordinal = 4;

            unsigned diff = (target_weekday - first_wd + 7u) % 7u;
            day_num = 1 + diff + 7u * (ordinal - 1);
            break;
        }

        case schedule::last: {
            unsigned last_day_of_month = days_in_month(m_year, m_month);
            sys_days last = sys_days{year_month_day{m_year / m_month / day{last_day_of_month}}};
            weekday last_wd{last};

            unsigned diff = (last_wd - target_weekday + 7u) % 7u;
            day_num = last_day_of_month - diff;
            break;
        }
    }

    return year_month_day{m_year / m_month / day{day_num}};
}

}  // namespace meetup
