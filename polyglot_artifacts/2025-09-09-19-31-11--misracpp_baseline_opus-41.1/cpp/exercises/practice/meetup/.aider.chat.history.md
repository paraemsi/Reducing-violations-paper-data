std::string scheduler::meetup_day(int year, int month, week w, weekday wd) {
    // Helper function to get the day of week for a given date
    auto get_weekday = [](int year, int month, int day) -> int {
        std::tm time_in = {};
        time_in.tm_year = year - 1900;
        time_in.tm_mon = month - 1;
        time_in.tm_mday = day;
        std::mktime(&time_in);
        return time_in.tm_wday;
    };
    
    // Helper function to get the number of days in a month
    auto days_in_month = [](int year, int month) -> int {
        if (month == 2) {
            // Check for leap year
            if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) {
                return 29;
            }
            return 28;
        }
        if (month == 4 || month == 6 || month == 9 || month == 11) {
            return 30;
        }
        return 31;
    };
    
    int target_weekday = static_cast<int>(wd);
    int max_days = days_in_month(year, month);

        // Find the teenth day (13-19) that matches the weekday
        for (int day = 13; day <= 19; ++day) {
            if (get_weekday(year, month, day) == target_weekday) {
                std::ostringstream oss;
                oss << year << "-" 
                    << std::setfill('0') << std::setw(2) << month << "-"
                    << std::setfill('0') << std::setw(2) << day;
                return oss.str();
            }
        }
    } else if (w == week::last) {
        // Find the last occurrence of the weekday in the month
        for (int day = max_days; day >= max_days - 6; --day) {
            if (get_weekday(year, month, day) == target_weekday) {
                std::ostringstream oss;
                oss << year << "-" 
                    << std::setfill('0') << std::setw(2) << month << "-"
                    << std::setfill('0') << std::setw(2) << day;
                return oss.str();
            }
        }
    } else {
        // Find the nth occurrence of the weekday (first, second, third, fourth, fifth)
        int occurrence = static_cast<int>(w);
        int count = 0;
        
        for (int day = 1; day <= max_days; ++day) {

                        << std::setfill('0') << std::setw(2) << day;

            }
        }
    }
    
    // If no valid date found (shouldn't happen with valid input)
    return "";
}
