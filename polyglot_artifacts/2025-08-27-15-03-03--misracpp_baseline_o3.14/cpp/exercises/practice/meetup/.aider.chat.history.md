day(std::chrono::year year,
    std::chrono::month month,
    std::chrono::weekday weekday,
    schedule spec) {

    auto y = year;
    auto m = month;

    if (spec == schedule::teenth) {
        for (std::chrono::day d{13}; d <= std::chrono::day{19}; ++d) {
            std::chrono::year_month_day date{y, m, d};
            if (std::chrono::weekday{date} == weekday) {
                return date;
            }
        }

        std::chrono::year_month_day_last ymdl{y, std::chrono::month_day_last{m}};
        std::chrono::sys_days sys{ymdl};
        std::chrono::weekday wd_last{sys};

        auto diff = (wd_last - weekday).count();
        if (diff < 0) diff += 7;
        sys -= std::chrono::days{diff};
        return std::chrono::year_month_day{sys};
    } else {
        int n = 0;
        switch (spec) {
            case schedule::first:  n = 1; break;

            case schedule::third:  n = 3; break;

    using namespace boost::gregorian;

    } else if (spec == schedule::last) {

        date candidate{year_, month_, last_day};
        while (candidate.day_of_week() != weekday) {
            candidate -= days{1};
        }
        return candidate;
    } else {

        switch (spec) {
            case schedule::first:  n = 1; break;
            case schedule::second: n = 2; break;
            case schedule::third:  n = 3; break;
            case schedule::fourth: n = 4; break;
