std::chrono::year_month_day scheduler::day(std::chrono::weekday const& wd,
                                           schedule const& spec) const
{
    using std::chrono::day;
    using std::chrono::year_month_day;
    using std::chrono::weekday;
    using std::chrono::months;
    using std::chrono::sys_days;

    /* First day of the target month */
    year_month_day const first_ymd{m_ym / day{1U}};
    weekday const first_wd{first_ymd};

    if (spec == schedule::teenth) {
        for (std::uint32_t d = 13U; d <= 19U; ++d) {
            year_month_day const cur{m_ym / day{d}};
            if (weekday{cur} == wd) {
                return cur;
            }
        }
    } else if ((spec == schedule::first) || (spec == schedule::second) ||
               (spec == schedule::third) || (spec == schedule::fourth)) {
        /* Offset from the first of the month to the first occurrence of wd */
        std::uint32_t const delta =
            static_cast<std::uint32_t>((wd - first_wd).count());

        std::uint32_t week_index = 0U;
        if (spec == schedule::second) {
            week_index = 1U;
        } else if (spec == schedule::third) {
            week_index = 2U;
        } else if (spec == schedule::fourth) {
            week_index = 3U;
        }

        std::uint32_t const target_day =
            static_cast<std::uint32_t>(1U + delta + (week_index * 7U));

        return year_month_day{m_ym / day{target_day}};
    } else { /* schedule::last */
        /* First day of the following month */
        std::chrono::year_month const next_month = m_ym + months{1};
        year_month_day const first_next{next_month / day{1U}};

        /* sys_days provides arithmetic on days */
        sys_days const first_next_sys{first_next};
        sys_days       last_sys      = first_next_sys - std::chrono::days{1};

        weekday const last_wd{year_month_day{last_sys}};

        /* Difference to move backwards to the requested weekday */
        std::uint32_t const back_delta =
            static_cast<std::uint32_t>((last_wd - wd).count());

        last_sys -= std::chrono::days{static_cast<std::int64_t>(back_delta)};

        return year_month_day{last_sys};
    }

    /* Should never be reached – all enum values handled above */
    return first_ymd;
}
