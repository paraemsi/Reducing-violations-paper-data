/*  Helper: does ‘big’ contain ‘small’ as a contiguous subsequence?          */
static auto contains_subsequence(const std::vector<std::int32_t>& big,
                                 const std::vector<std::int32_t>& small) -> bool
{
    if (small.empty()) {
        return true;
    }

    return (std::search(big.cbegin(),
                        big.cend(),
                        small.cbegin(),
                        small.cend()) != big.cend());
}

/*  Public API: classify the relation between two lists                      */

           const std::vector<std::int32_t>& listB) -> Relationship
{
    if (listA == listB) {
        return Relationship::Equal;
    }

    if (contains_subsequence(listA, listB)) {
        return Relationship::Superlist;
    }
