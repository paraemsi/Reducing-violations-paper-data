#include <stdexcept>
#include <cmath>

namespace queen_attack {

queen::queen(std::pair<int, int> position)
    : m_position(position)
{
    if (position.first < 0 || position.first > 7 ||
        position.second < 0 || position.second > 7) {
        throw std::domain_error("Queen position must be on a 8x8 board");
    }
}

std::pair<int, int> queen::position() const {
    return m_position;
}

bool can_attack(const queen& white, const queen& black) {
    auto [w_col, w_row] = white.position();
    auto [b_col, b_row] = black.position();

    if (w_col == b_col && w_row == b_row) {
        throw std::domain_error("Queens cannot occupy the same position");
    }

    // Same row
    if (w_row == b_row) return true;
    // Same column
    if (w_col == b_col) return true;
    // Same diagonal
    if (std::abs(w_col - b_col) == std::abs(w_row - b_row)) return true;

    return false;
}

}  // namespace queen_attack

    : m_position(position)

    if (position.first < 0 || position.first > 7 ||

    }

std::pair<int, int> queen::position() const {
    return m_position;
}

bool can_attack(const queen& white, const queen& black) {
    auto [w_col, w_row] = white.position();
    auto [b_col, b_row] = black.position();

    if (w_col == b_col && w_row == b_row) {
        throw std::domain_error("Queens cannot occupy the same position");
    }

    // Same row
    if (w_row == b_row) return true;
    // Same column
    if (w_col == b_col) return true;
    // Same diagonal
    if (std::abs(w_col - b_col) == std::abs(w_row - b_row)) return true;

    return false;
}
