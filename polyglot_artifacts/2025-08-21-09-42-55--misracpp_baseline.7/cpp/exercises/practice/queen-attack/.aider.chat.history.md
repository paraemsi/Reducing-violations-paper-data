#### # Instructions

#### Given the position of two queens on a chess board, indicate whether or not they are positioned so that they can attack each other.  
####   
#### In the game of chess, a queen can attack pieces which are on the same row, column, or diagonal.

#### You are also able to answer whether the queens can attack each other.

namespace queen_attack {

queen::queen(int row, int col) : row_{row}, col_{col} {}

std::pair<int, int> queen::position() const {
    return {row_, col_};
}

bool can_attack(const queen& q1, const queen& q2) {
    auto [row1, col1] = q1.position();
    auto [row2, col2] = q2.position();

    // Same row
    if (row1 == row2) return true;
    // Same column
    if (col1 == col2) return true;
    // Same diagonal
    if (std::abs(row1 - row2) == std::abs(col1 - col2)) return true;

    return false;
}

}  // namespace queen_attack

queen::queen(int row, int col) : row_{row}, col_{col} {}

std::pair<int, int> queen::position() const noexcept {
    return {row_, col_};
}

bool can_attack(const queen& q1, const queen& q2) {
    auto [row1, col1] = q1.position();
    auto [row2, col2] = q2.position();

    // Same row
    if (row1 == row2) return true;
    // Same column
    if (col1 == col2) return true;
    // Same diagonal
    if (std::abs(row1 - row2) == std::abs(col1 - col2)) return true;

    return false;
}
