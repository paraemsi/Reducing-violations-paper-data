#### Given any two lists `A` and `B`, determine if:  
####   
#### - List `A` is equal to list `B`; or  
#### - List `A` contains list `B` (`A` is a superlist of `B`); or  
#### - List `A` is contained by list `B` (`A` is a sublist of `B`); or  
#### - None of the above is true, thus lists `A` and `B` are unequal  
####   
#### Specifically, list `A` is equal to list `B` if both lists have the same values in the same order.  
#### List `A` is a superlist of `B` if `A` contains a sub-sequence of values equal to `B`.  
#### List `A` is a sublist of `B` if `B` contains a sub-sequence of values equal to `A`.  
####   
#### Examples:  
####   
#### - If `A = []` and `B = []` (both lists are empty), then `A` and `B` are equal  
#### - If `A = [1, 2, 3]` and `B = []`, then `A` is a superlist of `B`  
#### - If `A = []` and `B = [1, 2, 3]`, then `A` is a sublist of `B`  
#### - If `A = [1, 2, 3]` and `B = [1, 2, 3, 4, 5]`, then `A` is a sublist of `B`  
#### - If `A = [3, 4, 5]` and `B = [1, 2, 3, 4, 5]`, then `A` is a sublist of `B`  
#### - If `A = [3, 4]` and `B = [1, 2, 3, 4, 5]`, then `A` is a sublist of `B`  
#### - If `A = [1, 2, 3]` and `B = [1, 2, 3]`, then `A` and `B` are equal  
#### - If `A = [1, 2, 3, 4, 5]` and `B = [2, 3, 4]`, then `A` is a superlist of `B`  
#### - If `A = [1, 2, 4]` and `B = [1, 2, 3, 4, 5]`, then `A` and `B` are unequal

namespace sublist {

/**
 * Relationship between two lists.
 */
enum class Type {
    Equal,
    Sublist,
    Superlist,
    Unequal
};

/**
 * Determine the relationship of list A to list B.
 *
 * @tparam T   element type (must be comparable with operator==)
 * @param a    first list
 * @param b    second list
 * @return     Type::Equal      if a == b
 *             Type::Sublist    if a is contained in b
 *             Type::Superlist  if b is contained in a
 *             Type::Unequal    otherwise
 */
template <typename T>
inline Type classify(const std::vector<T>& a, const std::vector<T>& b) {
    // Fast path: identical vectors
    if (a == b) {
        return Type::Equal;
    }

    // Helper lambda to test whether 'small' is a contiguous sub-sequence of 'big'
    auto is_sublist = [](const std::vector<T>& small, const std::vector<T>& big) -> bool {
        if (small.empty()) {
            return true;            // Empty list is sublist of every list
        }
        if (small.size() > big.size()) {
            return false;           // Can't be a sublist if longer
        }
        for (std::size_t i = 0; i <= big.size() - small.size(); ++i) {
            if (std::equal(small.begin(), small.end(), big.begin() + static_cast<long>(i))) {
                return true;
            }
        }
        return false;
    };

    if (is_sublist(a, b)) {
        return Type::Sublist;
    }
    if (is_sublist(b, a)) {
        return Type::Superlist;
    }
    return Type::Unequal;
}

}  // namespace sublist
